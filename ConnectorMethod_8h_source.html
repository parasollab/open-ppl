479 ConnectorMethod<MPTraits>::
480 DoNotCheck(AbstractRoadmapType* const _r, const VID _source, const VID _target)
481  const {
482  // Check that both VIDs are valid.
483  if(_source == INVALID_VID or _target == INVALID_VID) {
484  if(this->m_debug)
485  std::cout << "\t\t\tSkipping connection with invalid node "
486  << "(" << _source << ", " << _target << ")."
487  << std::endl;
488  return true;
489  }
490 
491  // Check for self-connection.
492  if(_source == _target and !m_selfEdges) {
493  if(this->m_debug)
494  std::cout << "\t\t\tSkipping self-connection "
495  << "(" << _source << ", " << _target << ")."
496  << std::endl;
497  return true;
498  }
499 
500  // Check for cached connection.
501  if(this->IsCached(_r, _source, _target)) {
502  if(this->m_debug)
503  std::cout << "\t\t\tSkipping cached failed connection "
504  << "(" << _source << ", " << _target << ")."
505  << std::endl;
506  return true;
507  }
508 
509  // Check if the edge already exists.
510  if(_r->IsEdge(_source, _target)) {
511  if(this->m_debug)
512  std::cout << "\t\t\tSkipping existing connection "
513  << "(" << _source << ", " << _target << ")."
514  << std::endl;
515  return true;
516  }
517 
518  // Check if the nodes are in the same connected component.
519  if(m_skipIfSameCC) {
520  auto ccTracker = _r->GetCCTracker();
521  if(!ccTracker)
522  throw RunTimeException(WHERE) << "A CCTracker is required for checking "
523  << "same CCs.";
524  const bool sameCC = ccTracker->InSameCC(_source, _target);
525  if(sameCC) {
526  if(this->m_debug)
527  std::cout << "\t\t\tSkipping connection "
528  << "(" << _source << ", " << _target << ")"
529  << " within the same connected component."
530  << std::endl;
531  return true;
532  }
533  }
534 
535  return false;
536 }

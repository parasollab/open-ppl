143  const VertexSet& _candidates, OutputIterator _out) {
144  auto dm = this->GetDistanceMetric(this->m_dmLabel);
145 
146  if(this->m_debug)
147  std::cout << "Checking for nearest "
148  << (this->m_k ? std::to_string(this->m_k) : "all")
149  << " neighbors with dm '" << this->m_dmLabel << "'."
150  << "\n\tQuery cfg: " << _cfg.PrettyPrint()
151  << std::endl;
152 
153  // Keep a max pq of the k best so far (so that we can quickly test and remove
154  // the farthest).
155  std::priority_queue<Neighbor> pq;
156 
157  for(const VID vid : _candidates) {
158  // Check for prior connection.
159  if(this->m_unconnected and this->DirectEdge(_r, _cfg, vid))
160  continue;
161 
162  // Get the candidate Cfg and check against connection to self.
163  const auto& node = _r->GetVertex(vid);
164  if(node == _cfg)
165  continue;
166 
167  // Get the distance from the query cfg to the candidate. If it is infinite,
168  // these configurations are not connectable.
169  const double distance = dm->Distance(node, _cfg);
170  if(std::isinf(distance))
171  continue;
172 
173  // Track the closest m_k neighbors.
174  if(!this->m_k or pq.size() < this->m_k)
175  pq.emplace(vid, distance);
176  else if(distance < pq.top().distance) {
177  pq.pop();
178  pq.emplace(vid, distance);
179  }
180  }
181 
182  if(this->m_debug)
183  std::cout << "\tFound " << pq.size() << " neighbors." << std::endl;
184 
185  // Write k closest to vector, sorted greatest to least distance.
186  std::vector<Neighbor> closest;
187  closest.reserve(pq.size());
188  while(!pq.empty()) {
189  closest.push_back(pq.top());
190  pq.pop();
191  }
192 
193  // Write to output iterator in reverse order.
194  std::copy(closest.rbegin(), closest.rend(), _out);
195 }

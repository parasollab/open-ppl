490 CollisionDetectionValidity<MPTraits>::
491 IsInCollision(CDInfo& _cdInfo, const GroupCfg& _cfg, const std::string& _caller) {
492  const bool allInfo = _cdInfo.m_retAllInfo;
493  _cdInfo.ResetVars(allInfo);
494 
495  auto group = _cfg.GetGroupRoadmap()->GetGroup();
496  const auto& robots = group->GetRobots();
497 
498  if(this->m_debug)
499  std::cout << "Checking robot group '" << group->GetLabel()
500  << "' for collision."
501  << "\n\tCfg: " << _cfg.PrettyPrint()
502  << std::endl;
503 
504  bool collision = false;
505 
506  // Check the group for self-collisions.
507  if(!m_ignoreSelfCollision) {
508  // Check each robot for self-collision.
509  for(Robot* const robot : robots) {
510  collision |= IsInSelfCollision(_cdInfo, robot->GetMultiBody(), _caller);
511 
512  // Early quit if we don't care for all collision info.
513  if(!allInfo and collision)
514  return true;
515  }
516 
517  // Check for inter-robot collisions within the group.
518  std::vector<Robot*> remainingRobots = robots;
519  while(remainingRobots.size() > 1) {
520  // Take the last remaining robot out of the set and check it against the
521  // others.
522  Robot* const r = remainingRobots.back();
523  remainingRobots.pop_back();
524  collision |= IsInInterRobotCollision(_cdInfo, r, remainingRobots, _caller);
525 
526  // Early quit if we don't care for all collision info.
527  if(!allInfo and collision)
528  return true;
529  }
530  }
531 
532  // Check the group for boundary containment.
533  for(Robot* const robot : robots) {
534  collision |= IsInBoundaryCollision(_cdInfo, _cfg.GetRobotCfg(robot));
535 
536  // Early quit if we don't care for all collision info.
537  if(!allInfo and collision)
538  return true;
539  }
540 
541 
542  // Check for obstacle collisions.
543  for(Robot* const robot : robots) {
544  collision |= IsInObstacleCollision(_cdInfo, robot->GetMultiBody(), _caller);
545 
546  // Early quit if we don't care for all collision info.
547  if(!allInfo and collision)
548  return true;
549  }
550 
551  // We can quit now if we aren't using inter-robot collision, or if this group
552  // has all robots in the problem.
553  const auto& allRobots = this->GetMPProblem()->GetRobots();
554  const size_t otherRobotCount = allRobots.size() - group->Size();
555  if(!m_interRobotCollision or otherRobotCount == 0)
556  return collision;
557 
558  // Find the set of robots which are not in this group.
559  std::vector<Robot*> otherRobots;
560  otherRobots.reserve(otherRobotCount);
561  for(const auto& robot : allRobots)
562  if(!group->VerifyRobotInGroup(robot.get()))
563  otherRobots.push_back(robot.get());
564 
565  // Create a clearance map to track clearance info with otherRobots.
566  CDClearanceMap clearanceMap;
567 
568  // Check for inter-robot collisions.
569  for(Robot* const robot : robots) {
570  collision |= IsInInterRobotCollision(_cdInfo, robot, otherRobots, _caller);
571 
572  // Early quit if we don't care for all collision info.
573  if(!allInfo and collision)
574  return true;
575 
576  // Update self-clearance.
577  clearanceMap.Merge(_cdInfo.m_clearanceMap);
578  }
579 
580  _cdInfo.m_clearanceMap = clearanceMap;
581 
582  return collision;
583 }

587 bool
588 CollisionDetectionValidity<MPTraits>::
589 IsInSelfCollision(CDInfo& _cdInfo, const MultiBody* const _multibody,
590  const std::string& _caller) {
591  // If the robot has only one body, then it cannot be in self-collision.
592  if(_multibody->GetNumBodies() == 1)
593  return false;
594 
595  const size_t numBody = _multibody->GetNumBodies();
596  const bool allInfo = _cdInfo.m_retAllInfo;
597  bool collision = false;
598 
599  for(size_t i = 0; i < numBody - 1; ++i) {
600  const Body* const body1 = _multibody->GetBody(i);
601  for(size_t j = i + 1; j < numBody; ++j) {
602  const Body* const body2 = _multibody->GetBody(j);
603 
604  // Check for ignoring adjacent links.
605  if(m_ignoreAdjacentLinks and body1->IsAdjacent(body2))
606  continue;
607 
608  // Check for ignoring siblings.
609  if(m_ignoreSiblingCollisions and body1->SameParent(body2))
610  continue;
611 
612  CDInfo cdInfo(allInfo);
613  const bool c = m_cdMethod->IsInCollision(body1->GetPolyhedron(),
614  body1->GetWorldTransformation(),
615  body2->GetPolyhedron(),
616  body2->GetWorldTransformation(),
617  cdInfo);
618  collision |= c;
619  this->GetStatClass()->IncNumCollDetCalls(m_cdMethod->GetName(), _caller);
620 
621  if(this->m_debug and c)
622  std::cout << "\tSelf-collision detected between bodies " << i << " and "
623  << j << "." << std::endl;
624 
625  // Retain minimum distance information.
626  if(cdInfo < _cdInfo) {
627  _cdInfo = cdInfo;
630  _cdInfo.m_collidingObstIndex = -1;
631  }
632 
633  // If we don't need all information, terminate on the first collision.
634  if(!allInfo and collision)
635  return true;
636  }

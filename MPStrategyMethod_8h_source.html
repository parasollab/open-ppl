342  auto startConstraint = this->GetTask()->GetStartConstraint();
343  if(!startConstraint)
344  return INVALID_VID;
345 
346  // If we have no start boundary, there is also nothing to do.
347  auto startBoundary = startConstraint->GetBoundary();
348  if(!startBoundary)
349  return INVALID_VID;
350 
351  MethodTimer mt(this->GetStatClass(),
352  this->GetNameAndLabel() + "::GenerateStart");
353 
354  // Determine which sampler to use.
355  const auto& samplerLabel = m_querySampler.empty() ? _samplerLabel
356  : m_querySampler;
357 
358  if(this->m_debug)
359  std::cout << "Generating start configuration with sampler " << samplerLabel
360  << ":"
361  << std::endl;
362 
363  // Generate a valid start configuration. We will try up to 100 times before
364  // giving up and declaring failure.
365  std::vector<CfgType> cfgs;
366  auto sampler = this->GetSampler(samplerLabel);
367  sampler->Sample(1, 100, startBoundary, std::back_inserter(cfgs));
368 
369  // Throw an error if we failed to generate a single configuration.
370  if(cfgs.empty())
371  throw RunTimeException(WHERE) << "Could not generate valid start "
372  << "configuration in boundary "
373  << *startBoundary
374  << " with sampler '" << samplerLabel
375  << "'.";
376 
377  // Add the configuration to the roadmap.
378  const auto& start = cfgs.front();
379  auto g = this->GetRoadmap();
380  const VID vid = g->AddVertex(start);
381 
382  if(this->m_debug)
383  std::cout << "\tVID " << vid << " at " << start.PrettyPrint()
384  << std::endl;
385 
386  // Ensure the configuration satisfies the boundary we just used to sample it.
387  if(!this->GetTask()->EvaluateStartConstraints(g->GetVertex(vid)))
388  throw RunTimeException(WHERE) << "Sampled configuration from a start "
389  << "boundary, but task claims it doesn't satisfy:"
390  << "\n\tBoundary: " << *startBoundary
391  << "\n\tCfg: " << start.PrettyPrint()
392  << "\n\tFull cfg: " << start;
393  // Ensure the goal tracker recognized this configuration.
394  if(!this->GetGoalTracker()->GetStartVIDs().count(vid))
395  throw RunTimeException(WHERE) << "Added VID " << vid << " as a start "
396  << "node, but GoalTracker didn't log it.";
397

410  return {};

412  MethodTimer mt(this->GetStatClass(),

414

417  : m_querySampler;
418  auto sampler = this->GetSampler(samplerLabel);
419 
420  if(this->m_debug)
421  std::cout << "There are " << goalConstraints.size()
422  << " goals in this problem."
423  << "\nGenerating goal configurations with sampler " << samplerLabel
424  << ":"
425  << std::endl;
426 
427  // Generate a valid goal configuration for each constraint. We will try up to
428  // 100 times before giving up and declaring failure.
429  std::vector<VID> vids;
430  std::vector<CfgType> cfgs;
431  for(size_t i = 0; i < goalConstraints.size(); ++i) {
432  // Get the boundary for this goal.
433  auto goalBoundary = goalConstraints[i]->GetBoundary();
434  if(!goalBoundary)
435  throw RunTimeException(WHERE) << "Goal " << i
436  << " produced a null boundary.";
437 
438  // Sample a configuration.
439  cfgs.clear();
440  sampler->Sample(1, 100, goalBoundary, std::back_inserter(cfgs));
441 
442  // Ensure we got one.
443  if(cfgs.empty())
444  throw RunTimeException(WHERE) << "Could not generate valid goal " << i
445  << " configuration in boundary "
446  << *goalBoundary
447  << " with sampler '"
448  << samplerLabel << "'.";
449 
450  // Add it to the roadmap.
451  auto& goal = cfgs.front();
452  auto g = this->GetRoadmap();
453  const VID vid = g->AddVertex(goal);
454 
455  vids.push_back(vid);

457  if(this->m_debug)

459  << std::endl;
460

463  if(!this->GetTask()->EvaluateGoalConstraints(g->GetVertex(vid), i))

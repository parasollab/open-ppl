737  MethodTimer mt(stats, this->GetNameAndLabel() + "::AddQuery");

742  queryPoints.push_back(start);

748  for(const VID vid : queryPoints) {

755  LPOutput<MPTraits> lp;
756  for(const Neighbor& n : neighbors) {
757  // Try connection.
758  const bool success = AttemptConnection(cfg, r->GetVertex(n.target), lp);
759 
760  // Skip failures.
761  if(!success)
762  continue;
763  connected = true;
764  if(this->m_debug)
765  std::cout << "\tConnected to " << n.target << "."
766  << std::endl;
767

769  r->AddEdge(vid, n.target, lp.m_edge);
770

772  const LocalComponentDescriptor d = FindLocalComponent(n.target);
773  VertexSet& vids = d.bridge
774  ? GetBridge(d)
775  : GetLocalComponent(d);
776  vids.insert(vid);
777

785  if(connected)
786  continue;
787

789  // If so, start a new local component for it.
790  const bool inRegion = CheckReachedRegion(vid, {vid});
791  if(inRegion)
792  continue;
793

795  // or connect.

797  }

1403 template <typename MPTraits>
1404 typename DynamicRegionsPRM<MPTraits>::LocalComponentDescriptor
1405 DynamicRegionsPRM<MPTraits>::
1406 MergeLocalComponents(const LocalComponentDescriptor& _d1,
1407  const LocalComponentDescriptor& _d2) {
1408  // Assert that the edge descriptor IDs match, or we're looking at local ccs
1409  // covering different edges.
1410  const bool edgeIdsOk = _d1.edgeDescriptor.id() == _d2.edgeDescriptor.id();
1411  if(!edgeIdsOk)
1412  throw RunTimeException(WHERE) << "Descriptors point to different edge IDs "
1413  << _d1 << ", " << _d2 << ".";
1414 
1415  // Define a function for merging bridges.
1416  auto mergeBridges = [this](const LocalComponentDescriptor& _d1,
1417  const LocalComponentDescriptor& _d2) {
1418  // Get both vertex sets.
1419  VertexSet& v1 = this->GetBridge(_d1),
1420  & v2 = this->GetBridge(_d2);
1421 
1422  // Merge the smaller bridge into the larger one.
1423  if(v1.size() > v2.size()) {
1424  v1.insert(v2.begin(), v2.end());
1425  this->m_bridges[this->GetTask()->GetRobot()][_d2.edgeDescriptor].erase(_d2.representative);
1426  return _d1;
1427  }
1428  else {
1429  v2.insert(v1.begin(), v1.end());
1430  this->m_bridges[this->GetTask()->GetRobot()][_d1.edgeDescriptor].erase(_d1.representative);
1431  return _d2;
1432  }
1433  };
1434 
1435  // Define a function for same-side merges.
1436  auto sameSideMerge = [this](const LocalComponentDescriptor& _save,
1437  const LocalComponentDescriptor& _merge) {
1438  // Get the vertex sets.
1439  VertexSet& save = this->GetLocalComponent(_save),
1440  & merge = this->GetLocalComponent(_merge);
1441 
1442  // Merge the smaller set into the larger, move result to save if necessary.
1443  if(save.size() >= merge.size())
1444  save.insert(merge.begin(), merge.end());
1445  else {
1446  merge.insert(save.begin(), save.end());
1447  save = std::move(merge);
1448  }
1449 
1450  this->m_localComponents[this->GetTask()->GetRobot()][_merge.edgeDescriptor].erase(_merge.representative);
1451  this->m_regions[this->GetTask()->GetRobot()][_merge.edgeDescriptor].erase(_merge.representative);
1452  return _save;
1453  };
1454 
1455  // Check if we are merging any bridges.
1456  const size_t numBridges = _d1.bridge + _d2.bridge;
1457  switch(numBridges) {
1458  // If both are bridges, merge into the larger one.
1459  case 2:
1460  {
1461  if(this->m_debug)
1462  std::cout << "\t\tMerging bridges " << _d1 << " and " << _d2 << "."
1463  << std::endl;
1464  return mergeBridges(_d1, _d2);
1465  }
1466  // If one is a bridge, promote the other and then merge bridges.
1467  case 1:
1468  {
1469  if(this->m_debug)
1470  std::cout << "\t\tMerging bridge " << (_d1.bridge ? _d1 : _d2)
1471  << " and component " << (_d1.bridge ? _d2 : _d1) << "."
1472  << std::endl;
1473  return _d1.bridge ? mergeBridges(_d1, PromoteLocalComponent(_d2))
1474  : mergeBridges(_d2, PromoteLocalComponent(_d1));
1475  }
1476  // Handle merging two non-bridges.
1477  case 0:
1478  {
1479  // If the components are from either side, we've made a bridge. Promote both
1480  // components and merge.
1481  const bool sameSide = _d1.edgeDescriptor == _d2.edgeDescriptor;
1482  if(!sameSide) {
1483  if(this->m_debug)
1484  std::cout << "\t\tMerging components " << _d1 << " and " << _d2
1485  << " into a new bridge."
1486  << std::endl;
1487  return mergeBridges(PromoteLocalComponent(_d1),
1488  PromoteLocalComponent(_d2));
1489  }
1490 
1491  if(this->m_debug)
1492  std::cout << "\t\tMerging components " << _d1 << " and " << _d2
1493  << " on the same side."
1494  << std::endl;
1495 
1496  // The components are from the same side. Find which component's region is
1497  // further along and keep that one.
1498  const ExpansionRegion* r1 = GetExpansionRegion(_d1),
1499  * r2 = GetExpansionRegion(_d2);
1500  // We may get null regions if the component is on a low-clearance edge. In
1501  // that case, we'll save the component with no region because the other
1502  // one will eventually hit the low-clearance point.
1503  const bool saveR1 = (r1 and r2 and r1->edgeIndex > r2->edgeIndex)
1504  or (!r1 and r2);
1505  return saveR1 ? sameSideMerge(_d1, _d2)
1506  : sameSideMerge(_d2, _d1);
1507  }
1508  // We really can't get here...
1509  default:
1510  throw RunTimeException(WHERE) << "Unreachable state.";
1511  }
1512 }

1600 std::pair<std::vector<double>,
1601  std::vector<typename DynamicRegionsPRM<MPTraits>::ExpansionRegion*>>
1602 DynamicRegionsPRM<MPTraits>::
1603 ComputeProbabilities() {
1604  auto stats = this->GetStatClass();
1605  MethodTimer mt(stats, this->GetNameAndLabel() + "::ComputeProbabilities");
1606 
1607  // Sum all weights of all current regions and collect them into a vector.
1608  double totalWeight = 0.;
1609  std::vector<ExpansionRegion*> regions;
1610  for(auto& edAndMap : m_regions.at(this->GetTask()->GetRobot())) {
1611  for(auto& representativeAndRegion : edAndMap.second) {
1612  ExpansionRegion* const region = &representativeAndRegion.second;
1613  totalWeight += region->GetWeight();
1614  regions.push_back(region);
1615  }
1616  }
1617 
1618  // Compute the probabilities for the current regions.
1619  std::vector<double> probabilities;
1620  probabilities.reserve(regions.size() + 1);
1621 
1622  // Compute the explore probability (added to all regions).
1623  const double explore = m_explore / (regions.size() + 1);
1624 
1625  // Compute the total probability (exploit + explore).
1626  for(const ExpansionRegion* const region : regions) {
1627  const double exploit = totalWeight > 100 * std::numeric_limits<double>::epsilon()
1628  ? (1 - m_explore) * region->GetWeight() / totalWeight
1629  : 0;
1630 
1631  probabilities.emplace_back(exploit + explore);
1632  }
1633 
1634  // Add a null pointer and probability for the full environment.
1635  regions.push_back(nullptr);
1636  probabilities.emplace_back(explore);
1637 
1638  return {probabilities, regions};
1639 }

1675  std::cout << "\tClearance " << clearance << " < 0 too low, deleting."

1681 
1682  InitializeLowClearanceMap();

1684

1686  }
1687 
1688  if(this->m_debug)
1689  std::cout << "\tRegion reached the end of its edge."
1690  << "\n\tTrying to connect new VIDs to outbound local components."
1691  << "\n\t\tVIDs: " << _newVIDs
1692  << std::endl;
1693 
1694  // Define a function for trying to connect this region's local component to
1695  // those rooted at the edge target.
1696  VertexSet connectedVIDs;
1697  auto connector = [this, &_newVIDs, &connectedVIDs](
1698  const SkeletonEdgeDescriptor _ed, const std::map<VID, VertexSet>& _ccs,
1699  const bool _bridges) {
1700  if(this->m_debug)
1701  std::cout << "\tTrying " << _ccs.size()
1702  << (_bridges ? " bridges " : " components ")
1703  << "on edge (" << _ed.id() << "|" << _ed.source() << ","
1704  << _ed.target() << ")..."
1705  << std::endl;
1706 
1707  auto r = this->GetRoadmap();
1708 
1709  // Check each CC for connections.
1710  bool connected = false;
1711  for(const auto& cc : _ccs) {
1712  const VID representative = cc.first;
1713  const LocalComponentDescriptor d{_ed, representative, _bridges};
1714  const VertexSet& vids = cc.second;
1715 
1716  // If we've already connected to any vertex in this CC, we don't need to
1717  // make more.
1718  if(HaveCommonVertex(connectedVIDs, vids)) {
1719  if(this->m_debug)
1720  std::cout << "\t\tSkipping component " << d << ", already connected."
1721  << std::endl;
1722  connected = true;
1723  continue;
1724  }
1725 
1726  // Check each of our new vids.
1727  for(const VID vid : _newVIDs) {
1728  // Make edges to the target component.
1729  const std::vector<EdgeOutput> edges = ConnectToComponent(
1730  r->GetVertex(vid), d);
1731 
1732  // If we made no edges, move on to the next VID.
1733  if(edges.empty())
1734  continue;
1735 
1736  // Add the connections to the roadmap and move on to the next CC.
1737  connected = true;
1738  for(const auto& edge : edges) {
1739  r->AddEdge(vid, edge.target, edge.weights);
1740  connectedVIDs.insert(edge.target);
1741  }
1742 
1743  if(this->m_debug)
1744  std::cout << "\t\t"
1745  << (edges.size() ? "Formed connections on"
1746  : "Failed to connect to")
1747  << " component " << d << "."
1748  << std::endl;
1749  break;
1750  }
1751  }
1752 
1753  return connected;
1754  };
1755 
1756  // Try to connect to any local components rooted at this vertex.
1757  bool connected = false;
1758  const SkeletonVertexIterator vit = m_skeleton.find_vertex(
1759  (*_r->edgeIterator).target());
1760  for(auto eit = vit->begin(); eit != vit->end(); ++eit) {
1761  const SkeletonEdgeDescriptor ed = eit->descriptor();
1762  const auto& components = m_localComponents[this->GetTask()->GetRobot()][ed];
1763  const auto& bridges = m_bridges[this->GetTask()->GetRobot()][ed];
1764 
1765  connected |= connector(ed, components, false);
1766  connected |= connector(ed, bridges, true);
1767  }
1768 
1769  // If we didn't connect, initialize new regions on the outbound edges of the
1770  // target.
1771  if(!connected) {
1772  if(this->m_debug)
1773  std::cout << "\tRegion did not connect to any components at the target."
1774  << "\tSparking new regions on the outbound edges."
1775  << std::endl;
1776  const VID newRepresentative = *_newVIDs.begin();
1777  for(auto eit = vit->begin(); eit != vit->end(); ++eit)
1778  MakeLocalComponent({eit->descriptor(), newRepresentative, false}, _newVIDs);
1779  }
1780 
1781  // Promote component to bridge (also removes the region).
1782  PromoteLocalComponent(d);
1783  UpdateEdgeConnectivity(d.edgeDescriptor);
1784 
1785  return false;
1786 }

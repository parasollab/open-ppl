391  if(merged.count(i))

393

395 
396  size_t zero = 0;
397  size_t min = std::max(zero,std::min(interval1.min,interval1.min - (firstTime ? buffer : 0)));
398  size_t max = std::max(interval1.max,interval1.max + (firstTime ? buffer : 0));
399 
400  for(size_t j = i+1; j < copyIntervals.size(); j++) {
401

406  continue;

428  return _r1.min < _r2.min;
429  }
430  };

448  max = std::min(iter->min - 1,iter->min);

230 LocalPlannerMethod<MPTraits>::
231 BlindPath(const std::vector<CfgType>& _waypoints,
232  const double _posRes, const double _oriRes) {
233  // Blind local-plan between each intermediate,
234  bool first = true;
235  std::vector<CfgType> out;
236  LPOutput<MPTraits> lpOutput;
237  for(auto iter = _waypoints.begin(); iter + 1 != _waypoints.end(); ++iter) {
238  // If this isn't the first configuration, then its an intermediate and must
239  // be added to the path.
240  if(!first)
241  out.push_back(*iter);
242  else
243  first = false;
244 
245  // Reconstruct the resolution-level edge and append it to the output.
246  lpOutput.Clear();
247  IsConnected(*iter, *(iter + 1), &lpOutput, _posRes, _oriRes, false, true);
248  out.insert(out.end(), lpOutput.m_path.begin(), lpOutput.m_path.end());
249  }
250  return out;
251 }

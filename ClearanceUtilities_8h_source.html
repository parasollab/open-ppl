502 
503  if(!_cfg.InBounds(_b)) {

534  << std::endl;
535  return false;
536  }

539  VDAddTempCfg(_clrCfg, true);
540

550 ClearanceUtility<MPTraits>::
551 GetNearestVertexWitness(CfgType& _cfg, CDInfo& _cdInfo,
552  const Boundary* const _b) {
553  //This function places the nearest vertex of the environment and the nearest
554  // robot vertex into _cdInfo.
555  //Returns the validity of _cfg provided.
556 
557  // Setup Validity Checker
558  auto vcm = this->GetValidityChecker(m_vcLabel);
559 
560  _cdInfo.ResetVars(true);//true so we get all data back
561 
562  const string callee = this->GetNameAndLabel() + "::GetNearestVertexWitness";
563 
564  // Checking the validity here populates the _cdInfo object.
565  const bool initValidity = vcm->IsValid(_cfg, _cdInfo, callee); // Initial Validity
566 
567  //If initially invalid, then we are doing penetration. Return a negative
568  // distance in order to indicate this.
569  if(!initValidity)
570  _cdInfo.m_minDist = -_cdInfo.m_minDist;
571 
572  // If not using the bbx, we have the closest obstacle point already.
573 
574  if(this->m_useBBX) {
575  // CfgType is now know as good, get BBX and ROBOT info
576  auto multiBody = _cfg.GetMultiBody();
577 
578  // Find closest point between robot and bbx, set if less than min dist
579  // from obstacles
580  for(size_t m = 0; m < multiBody->GetNumBodies(); ++m) {
581  const GMSPolyhedron& poly = multiBody->GetBody(m)->GetWorldPolyhedron();
582  for(size_t j = 0; j < poly.GetVertexList().size(); ++j) {
583  const double clr = _b->GetClearance(poly.GetVertexList()[j]);
584  if(clr < _cdInfo.m_minDist) {
585  _cdInfo.m_robotPoint = poly.GetVertexList()[j];
586  _cdInfo.m_objectPoint = _b->GetClearancePoint(poly.GetVertexList()[j]);
587  _cdInfo.m_minDist = clr;
588  _cdInfo.m_nearestObstIndex = -1;
589  }
590  }
591  }
592  }
593 
594  return initValidity;
595 }

598 template<class MPTraits>
599 bool
600 ClearanceUtility<MPTraits>::
601 AdjustWitnessToEnsureValidity(const CfgType& _cfg, const bool _initValidity,
602  CfgType& _witnessCfg, const Boundary* const _b,
603  const bool _useOppValidityWitness) {
604  //_cfg is the sample configuration of _initValidity. _witnessCfg is its
605  // initial witness, which is just the translated _cfg from its nearest vertex
606  // to the nearest obstacle vertex.
607  Environment* env = this->GetEnvironment();
608  // Setup Validity Checker
609  auto vcm = this->GetValidityChecker(m_vcLabel);
610 
611  //See if the current _witnessCfg is a passing witness.
612  const string callee = this->GetNameAndLabel()+"::AdjustWitnessToEnsureValidity";
613  bool currValidity = vcm->IsValid(_witnessCfg, callee);
614  if(this->m_useBBX)
615  currValidity = currValidity && _witnessCfg.InBounds(_b);
616 
617  const bool witnessIsSameValidity = (_initValidity == currValidity);
618  bool passed = _useOppValidityWitness ? !witnessIsSameValidity :
619  witnessIsSameValidity;
620 
621  if(!passed) {
622  auto robot = _cfg.GetRobot();
623  //Get the increment:
624  CfgType incr(robot);
625  int nTicks; //This is used as the maximum number of adjustments possible.
626  incr.FindIncrement(_cfg, _witnessCfg, &nTicks,
627  env->GetPositionRes(), env->GetOrientationRes());
628  nTicks = max(nTicks,100); // Minimum allowable ticks bound is 100 for now.
629  int count = 0;
630  while(!passed) {
631  //All of the cases boil down to this condition wrt which direction to move
632  // (think about it in terms of the _useOppValidityWitness value, if it's
633  // set to true, then the only condition that could be in here is if they
634  // have the same validity, otherwise it's false and they are opposite).
635  if(witnessIsSameValidity)
636  _witnessCfg += incr;//Push out until validity changes.
637  else
638  _witnessCfg -= incr;//Pull cfg back until validity changes.
639 
640  //Note: a potential optimization is to only do a validity check against
641  // the obstacle originally colliding with the witness, as that shouldn't
642  // change when adjusting. It would only help certain cases though.
643  currValidity = vcm->IsValid(_witnessCfg, callee);
644  const bool inBounds = _witnessCfg.InBounds(_b);
645  if(this->m_useBBX)
646  currValidity = currValidity && inBounds;
647  passed = _useOppValidityWitness ? (_initValidity != currValidity) :
648  (_initValidity == currValidity);
649 
650  //If we go out of bounds while adjusting a witness, call it a failure.
651  // (Note that an acceptable OOB witness would have already been caught)
652  // Checking not passed here is very important!
653  if(!inBounds && !passed) {
654  if(this->m_debug)
655  cout << callee << ": Fell out of BBX after " << count << " steps, "
656  "returning false" << endl;
657  return false;
658  }
659  if(++count >= nTicks) {
660  if(this->m_debug)
661  std::cout << callee << ": Tried more than nTicks = " << nTicks <<
662  " when adjusting witness, returning false" << std::endl;
663  return false;
664  }
665  }
666  }
667 
668  return passed;
669 }

781  //or not free->free or iterates n times. Within it, it loops through all
782  //rays each time, moving outward along these and checking that condition.
783  //m_maxRayIterations is calculated by maxRayMagnitude/rayTickResolution

790  i += 1;

793

795  CDInfo tmpInfo;

799

801  // we are out of bounds and initially invalid, this ray can never lead
802  // to a successful witness of opposite validity, so remove the ray.
803  //Note: this is not taking into account whether or not m_useBBX is true.
804  // If doing the same-validity witness, then we still will count the tick
805  // before going out of bounds as the nearest witness, since it still must
806  // correspond to an obstacle's boundary being crossed.
807  if(!inBounds && !_initValidity && _useOppValidityWitness) {
808  if(this->m_debug)

810

812  if(_rays.size() == 1) {
813  if(this->m_debug)
814  cout << "Last valid ray just failed, Returning false" << endl;
815  return false;
816  }
817  else {
818  //Swap the ray leading to an invalid OOB cfg to the back of the vector
819  // and then pop from back of vector:
820  if(rit != _rays.end() - 1) { //Don't swap if it's already the last ray
821  auto eit = _rays.end() - 1;
822  swap(*rit, *eit);
823  }
824 
825  //Remove the ray that cannot lead to a valid witness.
826  _rays.pop_back();
827  continue; //move to next ray (note: won't advance the iterator)
828  }
829  }
830  //If we are doing the same validity witness, then the same triggers for a
831  // witness still apply, except that the condition above should also
832  // trigger a witness instead of disqualifying a ray.
833  else if (!inBounds && !_initValidity && !_useOppValidityWitness) {
834  // Force a change in validity, since we have found the edge of the
835  // obstacle we are in and it happens to border the BBX boundary.
836  currValidity = !_initValidity;
837  }
838  //So now we know that we have a ray tick that is either OOB and was an
839  // initially valid sample, or we are in bounds and initial state is N/A.
840  else {
841  currValidity = vcm->IsValid(rit->m_tick, tmpInfo, callee);
842  if(m_useBBX) // Only consider bounds here if using it as obstacle
843  currValidity = inBounds && currValidity;
844  }
845 
846  if(this->m_debug)
847  cout << " (currValidity for direction " << distance(_rays.begin(), rit)
848  << " = " << currValidity << ")";
849 
850  //if validity state has changed, save the candidate and stop searching:
851  //Note that OOB is of the same invalid status here as in collision.
852  if(currValidity != _initValidity) {
853  candidateFound = true; // So the enveloping while loop will exit
854 
855  candidateCfg = rit->m_tick;
856  //tick back the current cfg only if using the same validity witness:
857  if(!_useOppValidityWitness)
858  candidateCfg -= rit->m_incr;
859 
860  //Note: this is pushing back the index of the candidate ray (what
861  // distance() returns) and the actual location that we have found. The
862  // location's validity is determined by the type of witness desired.
863  _candidate = make_pair(distance(_rays.begin(), rit), candidateCfg);
864 
865  //Quit the loop since the witness has been found.
866  break;
867  }
868  //Must increment the iterator here due to the flow of the for loop:
869  ++rit;
870 
871  }
872  }//end while (!stateChangedFlag && iterations < max)
873  //Check that we actually found a candidate
874  if(!candidateFound) {
875  if(this->m_debug)
876  std::cout << callee + "Returning false after not finding a witness "
877  "candidate" << std::endl;
878  return false;
879  }
880 
881  if(this->m_debug) {
882  cout << "\nDEBUG:: done stepping out along rays. Candidate = " << std::endl;
883  cout << "\tray " << _candidate.first << ": " << _candidate.second;
884 
885  CDInfo tmpInfo;
886  bool currValidity = vcm->IsValid(_candidate.second, tmpInfo, callee);
887  if(m_useBBX)
888  currValidity = (currValidity && _candidate.second.InBounds(_b));
889  cout << " (currValidity = " << currValidity << ")" << endl;
890  }
891 
892  // Ensure the candidate is valid:
893  // Note: this is something we could remove for optimization
894  if(!ValidateCandidate(_candidate, _rays, _b, _useOppValidityWitness))
895  return false;
896 
897  // This check should be impossible to trigger, but it's a good thing to ensure
898  if(candidateCfg == _sampledCfg) {
899  if(this->m_debug)
900  std::cout << callee + "returning false from _clrCfg == _cfg" << std::endl;
901  return false;
902  }
903 
904  CDInfo tmpInfo;
905  bool currValidity = vcm->IsValid(candidateCfg, tmpInfo, callee);
906  if(this->m_useBBX)
907  currValidity = currValidity && candidateCfg.InBounds(_b);
908 
909  //Final check to make sure the witness is of correct validity.
910  bool passed = _useOppValidityWitness ? (_initValidity != currValidity) :
911  (_initValidity == currValidity);
912  if(!passed) {
913  if(this->m_debug)
914  std::cout << callee + "returning false from candidate being of wrong "
915  "validity" << std::endl;
916  return false;
917  }
918 
919  //If we make it here, we have a successful witness.
920  return true;
921 }

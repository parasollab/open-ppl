268 FullCfgs(MPLibrary* const _lib) const {
269  if(m_vids.empty())
270  return std::vector<GroupCfg>();
271 
272  // Insert the first vertex.
273  std::vector<GroupCfg> out = {m_roadmap->GetVertex(m_vids.front())};
274 
275  for(auto it = m_vids.begin(); it + 1 < m_vids.end(); ++it) {
276  const VID source = *it,
277  target = *(it + 1);
278  const auto& edge = m_roadmap->GetEdge(source, target);
279 
280  // Insert intermediates between vertices. For assembly planning (skip edge):
281  // don't reconstruct the edge when it's for a part that has been placed off
282  // to the side, just use the two cfgs. This edge will just be (start, end).
283  if(!edge.SkipEdge()) {
284  auto e = m_roadmap->GetEdge(source,target);
285  auto edge = !e.GetIntermediates().empty() ? e.GetIntermediates()
286  : _lib->ReconstructEdge(m_roadmap, source, target);
287 
288  if(!edge.empty()) {
289  // Only grab the intermediate cfgs.
290  auto startIter = edge.begin();
291 
292  auto endIter = edge.end();
293 
294  out.insert(out.end(), startIter, endIter);
295  }

299  out.push_back(m_roadmap->GetVertex(target));
300  }

219 EET<MPTraits>::
220 Iterate() {
221  auto rdmp = this->GetRoadmap();
222 
223  // sample a sphere.
224  const CfgType target = this->SelectTarget();
225 
226  // find nearest neighbor and expand tree towards target.
227  // if expansion was successful, add to rdmp.
228  // nearest neighbor should not be goal vertex.
229  const VID nearestVID = this->FindNearestNeighbor(target, &rdmp->GetAllVIDs());
230  const VID newVID = this->ExpandTree(nearestVID, target);
231 
232  // adjust hyperparameters
233  if (newVID != INVALID_VID) { // expansion successful.
234  if (this->m_debug){
235  std::cout << "This expansion was successful." << std::endl;
236  }
237 
238  Point newPoint = rdmp->GetVertex(newVID).GetPoint();
239 
240  exploreExploitBias *= (1 - m_controlManipulation);
241 
242  // Iterate through the spheres from goal to root,
243  // figure out the farthest sphere that this sample is in.
244  // Then, advance.
245  std::vector<Sphere> spheresSeen;
246  VID iterationSphereVID = goalSphere.wavefrontVID;
247  while (iterationSphereVID != INVALID_VID) { // from goal to root.
248  auto iterationSphere = wavefrontExpansionSpheres[iterationSphereVID];
249  Point iterationSphereCenter = iterationSphere.center;
250 
251  if (Distance(iterationSphereCenter, newPoint) < iterationSphere.radius) {
252  if (this->m_debug) {
253  std::cout << "This sample is in sphere "
254  << iterationSphere.wavefrontVID << std::endl;
255  }
256 
257  // Update current sphere (SS.size() = 0 means goal sphere) and explore params
258  currentSphere = spheresSeen.size() == 0 ? iterationSphere : spheresSeen.back();
259  exploreExploitBias = m_defaultExploreExploit;
260 
261  if (this->m_debug) {
262  std::cout << "New \"current\" Sphere has VID: "
263  << currentSphere.wavefrontVID << std::endl;
264  }
265  break;
266  }
267 
268  // sphere advancement operations.
269  spheresSeen.push_back(iterationSphere);
270  iterationSphereVID = iterationSphere.parentVID;
271  }
272 
273  // This is not in the pseudocode but is necessary for our implementation.
274  this->TryGoalExtension(newVID);
275 
276  } else { // expansion unsuccessful.
277  exploreExploitBias *= (1 + m_controlManipulation);
278  }
279 
280 
281  if (exploreExploitBias > 1) { // Backtrack to previous sphere.
282  currentSphere = wavefrontExpansionSpheres[currentSphere.parentVID];
283  exploreExploitBias = m_defaultExploreExploit;
284  }

304

306  std::vector<VID> goals = this->GenerateGoals(m_samplerLabel);
307  pGoal = rdmp->GetVertex(goals[0]).GetPoint();
308  for (VID goal : goals) { // We don't want to grow the goals. Hence remove them from rdmp.
309  rdmp->DeleteVertex(goal);

311

314  InsertSphereIntoPQ(pStart, INVALID_VID, sphereQueue);
315 
316  // in-loop variables

318 
319  // Grow WE!

321  Sphere s = sphereQueue.top();
322  sphereQueue.pop();

327  if (this->m_debug) {
328  std::cout << "Investigating " <<
329  "Sphere " << s.center[0] << " "
330  << s.center[1] << " "
331  << s.center[2] << " "
332  << "with radius " << s.radius
333  << " priority: " << s.priority
334  << " Added to WE." << std::endl
335  << " (There are " << sphereQueue.size() << " elments in the queue)."
336  << std::endl;
337  }
338  if (wavefrontRoot.isEmpty()){
339  wavefrontRoot = s;
340  }
341 
342  // add edge to WE
343  VID parentVID = s.parentVID;
344  wavefrontExpansion.AddEdge(parentVID, sVID);
345  if (this->m_debug) {

374  samples = _p.SampleSphereSurface(s.center, radius, m_nSphereSamples);

376  samples = _p.SampleSphereSurface(s.center, s.radius, m_nSphereSamples);

382  }
383

386  // check which sphere in tree to attach this one to.
387 
388  // not in pseudocode, but we don't want to insert

392  Sphere sphere = wavefrontExpansionSpheres[sphereVID];
393 
394  // outside existing spheres

397  sphereAddedToQueue = true;
398  }
399 
400  if (sphereAddedToQueue) { break; }
401  }
402  } // calculate sphere diameter for all pts for loop
403 
404  } // tree growth while loop
405 
406  if (this->m_debug) {
407  std::cout << "Wavefront expansion has "

409  << " nodes." << std::endl;

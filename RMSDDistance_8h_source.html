106 double
107 RMSDDistance<MPTraits>::
108 RMSD(vector<Vector3d> _x, vector<Vector3d> _y, int _dim) {
109  if((int)_x.size() < _dim || (int)_y.size() < _dim || _dim <= 0) {
110  cout << "Error in MyDistanceMetrics::RMSD, not enough data in vectors"
111  << endl;
112  exit(-1);
113  }
114 
115  //rmsd = sqrt( sum_of[(U*xn - yn)^2]/N )
116  //where U is the rotation that minimizes rmsd
117  //reference: B.Kabsch '78. Acta Cryst. (1978) A34 page 827-828
118 
119  //first step, remove any translation between x and y.
120  int n;
121  Vector3d sumx(0,0,0), sumy(0,0,0);
122  for(n=0; n<_dim; ++n) {
123  sumx = sumx + _x[n];
124  sumy = sumy + _y[n];
125  }
126  for(n=0; n<_dim; ++n) {
127  _x[n] = _x[n] - sumx/_dim;
128  _y[n] = _y[n] - sumy/_dim;
129  }
130 
131 
132  // now calc. e0 = 1/2*sum_of[xn^2 + yn^2]
133  double e0 = 0.0;
134  double r[3][3] = { {0, 0, 0}, {0, 0, 0}, {0, 0, 0}};
135  for(n=0; n<_dim; ++n) {
136  e0 += _x[n].normsqr() + _y[n].normsqr();
137  for(int i=0; i<3; ++i) {
138  for(int j=0; j<3; ++j)
139  r[i][j] += _y[n][i]*_x[n][j]; // *weight[n] if needed.
140  }
141  }
142  e0 /= 2;
143 
144  //let matrix r~*r = { a[0] d e
145  // d a[1] f
146  // e f a[2] };
147  //Now, decide this parameters.
148  //using a matrix here would be clearer, simply s = r.transpose()*R;
149  Vector3d col[3];
150  double a[3], d, e, f;
151  double detR; // determinant of r, we need its sign later.
152  for(int i=0; i<3; ++i) {
153  col[i] = Vector3d(r[0][i], r[1][i], r[2][i]);
154  a[i] = col[i].normsqr();
155  }
156  d = col[0]*col[1];
157  e = col[0]*col[2];
158  f = col[1]*col[2];
159  Vector3d col1X2 = col[1] % col[2];
160  detR = col[0]*col1X2;
161 
162  // now solve for the eigenvalues of the matrix, since we
163  // know we have three non-negative eigenvalue, we can directly
164  // solve a cubic equation for the roots...
165  // the equation looks like: z^3 + a2*z^2 + a1*z + a0 = 0
166  // in our case,
167  double a2 = -(a[0] + a[1] + a[2]);
168  double a1 = a[0]*a[1] + a[1]*a[2] + a[2]*a[0] - d*d - e*e - f*f;
169  double a0 = a[0]*f*f + a[1]*e*e + a[2]*d*d - a[0]*a[1]*a[2] - 2*d*e*f;
170 
171  // reference for cubic equation solution:
172  // http://mathworld.wolfram.com/CubicEquation.html
173  // following the symbols using there, define:
174  double q = (3*a1 - a2*a2) / 9;
175  double rr = (9.0*a2*a1 - 27.0*a0 - 2*a2*a2*a2) / 54;
176 
177  // if our case, since we know there are three real roots,
178  // so D = q^3 + r^2 <= 0,
179  double z1, z2, z3;
180  if(q == 0) { // which means three identical roots,
181  z1 = z2 = z3 = -a2/3;
182  }
183  else { // q < 0
184  double rootmq = sqrt(-q);
185  double ceta = acos(-rr/q/rootmq);
186  double cc3 = cos(ceta/3); // = cos(ceta/3)
187  double sc3 = sqrt(1-cc3*cc3); // = sin(ceta/3)
188  z1 = 2*rootmq*cc3 - a2/3;
189  z2 = rootmq*(-cc3 + sc3*1.7320508) - a2/3;
190  z3 = rootmq*(-cc3 - sc3*1.7320508) - a2/3;
191  }
192  if(z3 < 0) // small numercal error
193  z3 = 0;
194 
195  int sign = detR > 0 ? 1 : -1;
196  double ee = e0 - sqrt(z1) - sqrt(z2) - sqrt(z3)*sign;
197  if(ee<0) // small numercal error
198  return 0;
199 
200  // since ee = 1/2 * sum_of[(Uxn - yn)^2], so rmsd is:
201  double rmsd = sqrt(ee*2/_dim);
202 
203  return rmsd;
204 }

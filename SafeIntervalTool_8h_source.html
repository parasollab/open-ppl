208 IsSafe(const CfgType& _cfg, const double _timestep) {
209  // Configure _cfg's robot at _cfg.
210  auto robotMultiBody = _cfg.GetRobot()->GetMultiBody();
211  robotMultiBody->Configure(_cfg);
212 
213  // Get the valididty checker and make sure it has type
214  // CollisionDetectionValidity.
218  auto basevc = this->GetValidityChecker(m_vcLabel);
219  auto vc = dynamic_cast<CollisionDetectionValidityMethod<MPTraits>*>(basevc);
220 
221  // Compute the step number associated with _timestep.
222  const double timeRes = this->GetEnvironment()->GetTimeRes();
223  const size_t currentStep = std::lround(_timestep / timeRes);
224 
225  // Check this configuration against each dynamic obstacle.
226  const auto& obstacles = this->GetMPProblem()->GetDynamicObstacles();
227  for(const auto& obstacle : obstacles) {
228  auto obStart = obstacle.GetStartTime();
229 
230  if(currentStep < obStart)
231  continue;
232 
233  auto relativeStep = currentStep - obStart;
234 
235  // Determine the obstacle's position at the current timestep. If it is
236  // already done moving, use its last position.
237  const auto& path = obstacle.GetPath();
238  const size_t lastStep = path.size(),
239  //useStep = std::min(currentStep, lastStep -1);
240  useStep = std::min(relativeStep, lastStep -1);
241  const CfgType& position = path[useStep];
242 
243  if(_cfg.GetRobot() == position.GetRobot()) {
244  throw RunTimeException(WHERE) << "HOUSTON WE HAVE A PROBLEM" << std::endl;
245  }
246 
247  // Configure the obstacle at the current timestep.
248  auto obstacleMultiBody = obstacle.GetRobot()->GetMultiBody();
249  obstacleMultiBody->Configure(position);
250 
251  // If the obstacle is in collision with _cfg at _timestep, return false
252  CDInfo cdInfo;
253  if(vc->IsMultiBodyCollision(cdInfo, obstacleMultiBody, robotMultiBody,
254  this->GetNameAndLabel())) {
255  return false;
256  }
257  }

260  return true;

266 SafeIntervalTool<MPTraits>::
267 ComputeSafeIntervals(const std::vector<Cfg>& _cfgs) {
268  MethodTimer mt(this->GetStatClass(), "SafeIntervalTool::ComputeSafeIntervals");
269 
270  // If there are no dynamic obstacles, the safe interval is infinite.
271  const auto& obstacles = this->GetMPProblem()->GetDynamicObstacles();
272  if(obstacles.empty())
273  return {Range<double>(0, std::numeric_limits<double>::max())};
274 
275 // std::map<CfgType, size_t> finalResting;
276 
277  // Find the latest timestep in which a dynamic obstacle is still moving.
278  size_t timeFinal = 0;
279  for(auto& obstacle : obstacles) {
280  auto path = obstacle.GetPath();
281 
282  auto finish = path.size() + obstacle.GetStartTime();
283 
284  //if(obstacle.GetPath().size() > timeFinal) {
285  if(finish > timeFinal) {
286  timeFinal = finish;//obstacle.GetPath().size();
287  }
288  }
289 
290  // Determine all of the intervals for which it is safe to start following this
291  // set of configurations.
292  const double timeRes = this->GetEnvironment()->GetTimeRes();
293 
294  Intervals safeIntervals; // All intervals for this sequence.
295  Range<double>* currentInterval = nullptr; // The current interval under construction.
296 
297  // Try starting the sequence at each time step where dynamic obstacles are
298  // moving.
299  for(size_t tstep = 0; tstep <= timeFinal; ++tstep) {
300  const double startTime = tstep * timeRes;
301 
302  // Check if it is safe to start the sequence at startTime.
303  bool safe = true;
304  size_t i = 0;
305  for(const CfgType& cfg : _cfgs) {
306  safe &= IsSafe(cfg, startTime + i * timeRes);
307  if(!safe)
308  break;
309  ++i;
310  }
311  // If it is not safe to start the sequence from startTime, end the current
312  // safe interval, if any.
313  if(!safe) {
314  currentInterval = nullptr;
315  }
316  // Otherwise, we can include this startTime in the safe intervals.
317  else {
318  // Start a new interval if we don't have one already.
319  if(!currentInterval) {
320  safeIntervals.emplace_back(startTime, startTime);
321  currentInterval = &safeIntervals.back();
322  }
323 
324  // Expand the end of the current interval to include this startTime.
325  currentInterval->max = startTime;
326  }
327  }
328 
329  // If we still have a current interval, then the last interval was safe
330  // through the end of the dynamic obstacle motions. Extend it to the end of
331  // time.
332  if(currentInterval)
333  currentInterval->max = std::numeric_limits<double>::max();
334  else {
335  safeIntervals.emplace_back((timeFinal +1) * timeRes, std::numeric_limits<double>::max());
336  }
337 
338  return safeIntervals;
339 }

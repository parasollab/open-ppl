491 template <typename MPTraits>
492 void
493 LazyQuery<MPTraits>::
494 NodeEnhance() {
495  if(!m_numEnhance || m_edges.empty())
496  return;
497 
498  if(this->m_debug)
499  std::cout << "\tLazyQuery is enhancing nodes...\n\t Generated VIDs:";
500 
501  auto dm = this->GetDistanceMetric(m_enhanceDmLabel);
502  auto roadmap = this->GetRoadmap();
503 
504  for(size_t i = 0; i < m_numEnhance and !m_edges.empty(); ++i) {
505  // Pick a random edge from m_edges.
506  const size_t index = LRand() % m_edges.size();
507 
508  // Get its midpoint and a random ray.
509  const CfgType midpoint = (m_edges[index].first + m_edges[index].second) / 2.;
510  CfgType ray(midpoint.GetRobot());
511  ray.GetRandomRay(std::abs(GaussianDistribution(0, m_d)), dm);
512 
513  // Create an enhancement cfg by adding the random ray to the midpoint.
514  CfgType enhance = midpoint + ray;
515  enhance.SetLabel("Enhance", true);
516 
517  // If enchancement cfg is in bounds, add it to the roadmap and connect.
518  if(enhance.InBounds(this->GetEnvironment())) {
519  const VID newVID = roadmap->AddVertex(enhance);
520  for(auto& label : m_ncLabels)
521  this->GetConnector(label)->Connect(roadmap, newVID);
522  if(this->m_debug)
523  std::cout << " " << newVID;
524  }
525 
526  // Release the enhancement edge.
527  m_edges.erase(m_edges.begin() + index);
528  }
529 
530  if(this->m_debug)
531  std::cout << std::endl;
532 }

568  const CfgType& cfg1 = this->m_roadmap->GetVertex(_source),
569  & cfg2 = this->m_roadmap->GetVertex(_target);
570  if(!cfg1.IsLabel("Enhance") and !cfg2.IsLabel("Enhance")) {

575  }
576  }
577

579  // Also delete the reverse edge if it exists.
580  if(m_deleteInvalid) {
581  this->m_roadmap->DeleteEdge(_source, _target);
582  if(this->m_roadmap->IsEdge(_target, _source))
583  this->m_roadmap->DeleteEdge(_target, _source);

429  CfgType target(this->GetTask()->GetRobot());
430 
431  // Get the sampler and boundary.

434

438

443  const Boundary* const b = goalConstraints[index]->GetBoundary();

447  // implementation.

453  if(!this->m_querySampler.empty())

455 
456  if(this->m_debug)

479 template <typename MPTraits>
480 typename MPTraits::CfgType
481 BasicRRTStrategy<MPTraits>::
482 SelectDispersedTarget(const VID _v) {
483  MethodTimer mt(this->GetStatClass(),
484  this->GetNameAndLabel() + "::SelectDispersedTarget");
485 
486  // Get original cfg with vid _v and its neighbors
487  auto g = this->GetRoadmap();
488  const std::vector<VID> neighbors = g->GetChildren(_v);
489  const CfgType& originalCfg = g->GetVertex(_v);
490 
491  // Look for the best extension direction, which is the direction with the
492  // largest angular separation from any neighbor.
493  CfgType bestCfg(this->GetTask()->GetRobot());
494  double bestAngle = -MAX_DBL;
495  for(size_t i = 0; i < m_disperseTrials; ++i) {
496  // Get a random configuration
497  CfgType randCfg(this->GetTask()->GetRobot());
498  randCfg.GetRandomCfg(this->GetEnvironment());
499 
500  // Get the unit direction toward randCfg
501  CfgType randDir = randCfg - originalCfg;
502  randDir /= randDir.Magnitude();
503 
504  // Calculate the minimum angular separation between randDir and the
505  // unit directions to originalCfg's neighbors
506  double minAngle = MAX_DBL;
507  for(auto& vid : neighbors) {
508  const CfgType& neighbor = g->GetVertex(vid);
509 
510  // Get the unit direction toward neighbor
511  CfgType neighborDir = neighbor - originalCfg;
512  neighborDir /= neighborDir.Magnitude();
513 
514  // Compute the angle between randDir and neighborDir
515  double sum{0};
516  for(size_t j = 0; j < originalCfg.DOF(); ++j)
517  sum += randDir[j] * neighborDir[j];
518  const double angle = std::acos(sum);
519 
520  // Update minimum angle
521  minAngle = std::min(minAngle, angle);
522  }
523 
524  // Now minAngle is the smallest angle between randDir and any neighborDir.
525  // Keep the randDir that produces the largest minAngle.
526  if(bestAngle < minAngle) {
527  bestAngle = minAngle;
528  bestCfg = randCfg;
529  }
530  }
531 
532  return bestCfg;
533 }

837  TryGoalExtension(extended, _boundary);
838  }
839  else if(this->m_debug)
840  std::cout << "\tExtension did not make progress." << std::endl;

899 ConnectTrees(const VID _recentlyGrown) {

901  auto ccTracker = g->GetCCTracker();
902  const size_t ccCount = ccTracker->GetNumCCs();
903 
904  if(!m_growGoals or _recentlyGrown == INVALID_VID or ccCount == 1)
905  return;
906 
907  MethodTimer mt(this->GetStatClass(),
908  this->GetNameAndLabel() + "::ConnectTrees");
909 
910  // Get the configuration by value in case the graph's vertex vector gets
911  // re-allocated.
912  const CfgType qNew = g->GetVertex(_recentlyGrown);
913 
914  if(this->m_debug)
915  std::cout << "Trying to connect " << ccCount - 1 << " other trees "
916  << "to node " << _recentlyGrown << "."
917  << std::endl;
918 
919  // Try to connect qNew to each other CC.
920  VertexSet representatives = ccTracker->GetRepresentatives();
921  while(representatives.size())
922  {
923  // Get the next representative and remove it from the set.
924  const VID representative = *representatives.begin();
925  representatives.erase(representatives.begin());
926 
927  // If the new vertex and this are in the same CC, there is nothing to do.
928  const bool sameCC = ccTracker->InSameCC(_recentlyGrown, representative);
929  if(sameCC)
930  continue;
931 
932  // Get the CC associated with this representative.
933  const VertexSet* const cc = ccTracker->GetCC(representative);
934 
935  // Find nearest neighbor to qNew in the other tree.
936  const VID nearestVID = FindNearestNeighbor(qNew, cc);
937  if(nearestVID == INVALID_VID)
938  continue;
939 
940  // Try to extend from the other tree to qNew.
941  this->Extend(nearestVID, qNew, false);
942  }

463 AdaptiveRRT<MPTraits>::RewardGrowthMethod(double _r, std::string _s, GrowthSet& _gs){
464  if(this->m_debug)
465  std::cout << "Reward::" << _s << "::" << _r << std::endl;
466  // factor is e^(g * x_i' / K) where g is gamma, x_i' = x_i/p_i* where x_i is

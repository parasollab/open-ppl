213 PathType<MPTraits>::
214 Length() const {
215  // If the length is cached, we don't need to recompute.
216  if(m_lengthCached)
217  return m_length;
218  m_lengthCached = true;
219 
220  // Recompute the length by summing the edge weights.
221  m_length = 0;
222  for(auto start = m_vids.begin(); start + 1 < m_vids.end(); ++start) {
223  // Skip repeated vertices.
225  if(*start == *(start + 1))
226  continue;
227 
228  // Add this edge's weight to the sum.
229  const auto& edge = m_roadmap->GetEdge(*start, *(start + 1));
230  m_length += edge.GetWeight();
231  }

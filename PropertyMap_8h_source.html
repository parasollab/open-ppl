192 template <typename MPTraits>
193 PropertyMap<vector<double>,double>*
194 ClearanceAnnotatedSkeleton(MPBaseObject<MPTraits>* _mp, WorkspaceSkeleton* _ws,
195  bool _boundary = true) {
196  typedef typename MPTraits::CfgType CfgType;
197  auto clearanceMap = new PropertyMap<vector<double>,double>(_ws);
198 
199  auto g = _ws;
200  auto boundary = _mp->GetEnvironment()->GetBoundary();
201  auto vc = _mp->GetValidityChecker("pqp_solid");
202 
203  auto pointRobot = _mp->GetMPProblem()->GetRobot("point");
204 
205  // Function to compute clearance for input point _p.
206  auto getClearance = [&](const Point3d& _p) -> double {
207  // Check against obstacles using a point robot.
208  CfgType cfg(_p, pointRobot);
209  CDInfo cdInfo(true);
210  vc->IsValid(cfg, cdInfo, "Skeleton Clearance");
211 
212  // Check against boundary.
213  if(_boundary) {
214  const double boundaryClearance = boundary->GetClearance(_p);
215  if(boundaryClearance < cdInfo.m_minDist) {
216  cdInfo.m_objectPoint = boundary->GetClearancePoint(_p);
217  cdInfo.m_minDist = boundaryClearance;
218  }
219  }
220  // Return the minimum clearance.
221  return cdInfo.m_minDist;
222  };
223 
224  // Graph vertices clearance.
225  for(auto vit = g->begin(); vit != g->end(); ++vit)
226  clearanceMap->SetVertexProperty(vit->descriptor(),
227  getClearance(vit->property()));
228 
229  // Graph edges clearance.
230  for(auto eit = g->edges_begin(); eit != g->edges_end(); ++eit) {
231  vector<double> clearances;
232  for(auto pit = eit->property().begin(); pit < eit->property().end(); ++pit)
233  clearances.push_back(getClearance(*pit));
234  auto ed = WorkspaceSkeleton::ED(eit->source(), eit->target(),
235  eit->descriptor().id());
236  clearanceMap->SetEdgeProperty(ed,clearances);
237  }
238  return clearanceMap;
239 }

623 MPLibraryType<MPTraits>::
624 ReadXMLFile(const std::string& _filename) {
625  // Open the XML and get the root node.
626  XMLNode mpNode(_filename, "MotionPlanning");
627 
628  // Find the 'MPLibrary' node.
629  XMLNode* planningLibrary = nullptr;
630  for(auto& child : mpNode)
631  if(child.Name() == "Library")
632  planningLibrary = &child;
633 
634  // Throw exception if we can't find it.
635  if(!planningLibrary)
636  throw ParseException(WHERE) << "Cannot find MPLibrary node in XML file '"
637  << _filename << "'.";
638  ProcessXML(*planningLibrary);
639 }

641 template <typename MPTraits>
642 void
643 MPLibraryType<MPTraits>::
644 ProcessXML(XMLNode& planningLibraryNode) {
645 
646  // Parse the library node to set algorithms and parameters.
647  for(auto& child : planningLibraryNode)
648  ParseChild(child);
649 
650  // Ensure we have at least one solver.
651  if(m_solvers.empty())
652  throw ParseException(WHERE) << "Cannot find Solver node in XML node '.";
653 
654  // Set the seed to the first solver
655  SetSeed();
656 
657  // Print XML details if requested.
658  bool print = planningLibraryNode.Read("print", false, false, "Print all XML input");
659  if(print)
660  Print(cout);
661 
662  // Handle XML warnings/errors.
663  bool warnings = planningLibraryNode.Read("warnings", false, false, "Report warnings");
664  if(warnings) {
665  bool warningsAsErrors = planningLibraryNode.Read("warningsAsErrors", false, false,
666  "XML warnings considered errors");
667  planningLibraryNode.WarnAll(warningsAsErrors);
668  }

948 MPLibraryType<MPTraits>::
949 ReconstructEdge(RoadmapType* const _roadmap, const VID _source,
950  const VID _target, const double _posRes,
951  const double _oriRes) {
952  const auto& start = _roadmap->GetVertex(_source);
953  const auto& end = _roadmap->GetVertex(_target);
954  const auto& edge = _roadmap->GetEdge(_source, _target);
955  const auto& intermediates = edge.GetIntermediates();
956 
957  // Construct the set of start, intermediates, and end waypoints as needed.
958  auto waypoints = intermediates;
959  waypoints.insert(waypoints.begin(), start);
960  waypoints.push_back(end);
961 
962  // Check for intermediates. If there are any, we will use straight-line to
963  // recompute the path. Otherwise use the lp label if available, and fall back
964  // to straight-line if not (this will always happen with extenders).
965  const std::string lpLabel = !intermediates.size()
966  and !edge.GetLPLabel().empty()
967  ? edge.GetLPLabel()
968  : "sl";
969 
970  // Construct a resolution-level path along the recreated edge.
971  auto lp = this->GetLocalPlanner(lpLabel);
972  return lp->BlindPath(waypoints, _posRes, _oriRes);
973 }

1145  std::cout << "\n\nMPLibrary is solving with MPStrategyMethod labeled "
1146  << _solver.label << " using seed " << _solver.seed << "."
1147  << std::endl;
1148 
1149  // Save the original prefix for output files (base name).
1150  const std::string originalBaseFilename = m_problem->GetBaseFilename();
1151 
1152  // Use the solver node to set the base name for output files.
1153  std::string baseFilename = _solver.baseFilename;
1154 
1155  // If this task has a label, append it to the solver's output file name.
1156  if(m_groupTask) {
1157  if(!m_groupTask->GetLabel().empty())
1158  baseFilename += "." + m_groupTask->GetLabel();
1159  }
1160  else if(!m_task->GetLabel().empty())
1161  baseFilename += "." + m_task->GetLabel();
1162 
1163  // Remove spaces to keep file names nice.
1164  {
1165  auto newEnd = std::remove_if(baseFilename.begin(), baseFilename.end(),
1166  ::isspace);
1167  baseFilename.erase(newEnd, baseFilename.end());
1168  }
1169 
1170  // Set the output file locations.
1171  SetBaseFilename(GetMPProblem()->GetPath(baseFilename));
1172  GetStatClass()->SetAuxDest(GetBaseFilename());
1173 
1174  // Initialize vizmo debug if requested.
1175  if(_solver.vizmoDebug) {
1176  VDInit(GetBaseFilename() + ".vd");
1177  if(m_groupTask)
1178  VDTrackRoadmap(this->GetGroupRoadmap());
1179  else
1180  VDTrackRoadmap(this->GetRoadmap());
1181  }
1182 
1183  // Initialize the library's algorithms.
1184  SetSeed(_solver.seed);
1185  Initialize();
1186 
1187  // Reset the seed
1188  SetSeed(_solver.seed);
1189 
1190  GetMPStrategy(_solver.label)->operator()();
1191 
1192  // Close vizmo debug if necessary
1193  if(_solver.vizmoDebug)
1194  VDClose();
1195

1197  Uninitialize();

595 RewireTest(RoadmapType* const _r, const VID _vid,
596  const VID _currentParent, const double _currentCost,
597  const VID _potentialParent, const double _potentialParentCost,
598  OutputIterator<RoadmapType>* const _collision) noexcept {

600  const bool sameParent = _potentialParent == _currentParent;
601  if(sameParent) {
602  if(this->m_debug)
603  std::cout << "\t\tNot rewiring node " << _vid
604  << " which is already a child of " << _potentialParent << "."
605  << std::endl;
606  return RewireTestOutput();
607  }
608 
609  // Check for previous failures to generate this local plan.
610  const bool previouslyFailed = this->IsCached(_r, _potentialParent, _vid);
611  if(previouslyFailed) {
612  if(this->m_debug)
613  std::cout << "\t\tNot rewiring node " << _vid << " because the path ("
614  << _potentialParent << ", " << _vid
615  << ") was already found invalid."
616  << std::endl;
617  return RewireTestOutput();
618  }
619 
620  // Skip rewiring if the current path to _vid is better than the path to
621  // _potential parent.
622  const bool notBetterTo = m_objective(_potentialParentCost, _currentCost);
623  if(notBetterTo) {
624  if(this->m_debug)
625  std::cout << "\t\tNot rewiring node " << _vid << " because current cost "
626  << _currentCost << " is better than cost to " << _potentialParent
627  << " of " << _potentialParentCost << "."
628  << std::endl;
629  return RewireTestOutput();
630  }
631 
632  // If the total cost isn't better, do not rewire.
633  auto nf = this->GetNeighborhoodFinder(m_nfLabel);
634  auto dm = this->GetDistanceMetric(nf->GetDMLabel());
635  const CfgType& potentialParentCfg = _r->GetVertex(_potentialParent),
636  & cfg = _r->GetVertex(_vid);
637  const double distance = dm->Distance(potentialParentCfg, cfg),
638  potentialCost = _potentialParentCost + distance;
639  const bool notBetterThrough = m_objective(potentialCost, _currentCost);
640  if(notBetterThrough) {
641  if(this->m_debug)
642  std::cout << "\t\tNot rewiring node " << _vid << " because current cost "
643  << _currentCost << " is better than cost through "

645  << std::endl;
646  return RewireTestOutput();

725  std::cout << "\t\tNot rewiring node " << _vid << " because the path ("
726  << _potentialParent << ", " << _vid
727  << ") was already found invalid."
728  << std::endl;
729  return GroupRewireTestOutput(false, std::move(glpo), 0.0);
730  }
731 
732  // Skip rewiring if the current path to _vid is better than the path to
733  // _potential parent.
734  const bool notBetterTo = m_objective(_potentialParentCost, _currentCost);
735  if(notBetterTo) {
736  if(this->m_debug)
737  std::cout << "\t\tNot rewiring node " << _vid << " because current cost "
738  << _currentCost << " is better than cost to " << _potentialParent
739  << " of " << _potentialParentCost << "."
740  << std::endl;
741  return GroupRewireTestOutput(false, std::move(glpo), 0.0);
742  }
743 
744  // If the total cost isn't better, do not rewire.
745  auto nf = this->GetNeighborhoodFinder(m_nfLabel);
746  auto dm = this->GetDistanceMetric(nf->GetDMLabel());
747  const GroupCfgType& potentialParentCfg = _r->GetVertex(_potentialParent),
748  & cfg = _r->GetVertex(_vid);
749  const double distance = dm->Distance(potentialParentCfg, cfg),
750  potentialCost = _potentialParentCost + distance;
751  const bool notBetterThrough = m_objective(potentialCost, _currentCost);
752  if(notBetterThrough) {
753  if(this->m_debug)
754  std::cout << "\t\tNot rewiring node " << _vid << " because current cost "
755  << _currentCost << " is better than cost through "
756  << _potentialParent << " of " << potentialCost << "."
757  << std::endl;
758  return GroupRewireTestOutput(false, std::move(glpo), 0.0);
759  }
760 
761  // Create a local plan. If we fail, do not rewire.
762  auto lp = this->GetLocalPlanner(this->m_lpLabel);
763  auto env = this->GetEnvironment();
764  GroupCfgType collision;
765  GroupLPOutput<MPTraits> lpo(cfg.GetGroupRoadmap());
766 
767  const bool success = lp->IsConnected(potentialParentCfg, cfg, collision, &lpo,
768  env->GetPositionRes(), env->GetOrientationRes());
769 
770  if(!success) {
771  if(_collision)
772  *_collision = collision;
773  // Cache both directions since LPs must generate symmetric plans.
774  this->CacheFailedConnection(_r, _potentialParent, _vid);
775  this->CacheFailedConnection(_r, _vid, _potentialParent);
776  if(this->m_debug)
777  std::cout << "\t\tNot rewiring node " << _vid << " because the path ("
778  << _potentialParent << ", " << _vid
779  << ") is invalid."
780  << std::endl;
781  return GroupRewireTestOutput(false, std::move(glpo), 0.0);
782  }
783 
784  // Ensure that the computed LP cost is approximately equal to the distance
785  // metric cost (or none of this works because we didn't test the right set of
786  // neighbors in the first place).
790 #if 1
791  const double computedDistance = lpo.m_edge.first.GetWeight(),
792  tolerance = distance * .001;
793  const bool lpDmDisagreement = !nonstd::approx(computedDistance, distance,
794  tolerance);
795  if(lpDmDisagreement)
796  throw RunTimeException(WHERE) << "LP computed distance " << computedDistance
797  << " differs from DM value " << distance
798  << ". These must agree for RewireConnector "
799  << "to function properly.";
800 #endif
801 
802  // The cost through the potential parent is better.
803  if(this->m_debug)
804  std::cout << "\t\tNode " << _vid << " has a better parent "
805  << _potentialParent << " yielding path cost " << potentialCost
806  << "."
807  << std::endl;
808 
809  return GroupRewireTestOutput(true, std::move(lpo), potentialCost);
810 }

814 double
815 RewireConnector<MPTraits>::
816 ShortestPathWeight(const RoadmapType* const _r, const VID _vid)
817  const noexcept {
818  // Backtrack the path from _vid to its root by following the predecessor chain.
819  VID currentVID = _vid;
820  double pathWeight = 0;
821 
822  while(true) {
823  // Get the predecessor list and check if we're done.
824  const auto& predecessors = _r->GetPredecessors(currentVID);
825  switch(predecessors.size()) {
826  case 0:
827  // This node is a root. We've traced the whole path and can return the
828  // weight now.
829  return pathWeight;
830  case 1:
831  // This node has a parent. Continue processing.
832  break;
833  default:
834  // The graph is not a tree.
835  throw RunTimeException(WHERE) << "Node " << currentVID << " has "
836  << predecessors.size() << " parents and "
837  << "is therefore not a tree. "
838  << "RewireConnector only works for tree "
839  << "graphs."
840  << std::endl;
841  }
842 
843  // Add the edge weight to our path length and continue up the chain.
844  const VID parentVID = *predecessors.begin();
845 
846  const double edgeWeight = EdgeWeight(_r, parentVID, currentVID);
847  pathWeight = m_operator(pathWeight, edgeWeight);
848 
849  currentVID = parentVID;
850  }
851 
852  throw RunTimeException(WHERE) << "We've arrived at an unreachable state. "
853  << "https://xkcd.com/2200/";

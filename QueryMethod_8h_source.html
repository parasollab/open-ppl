252 QueryMethod<MPTraits>::
253 operator()() {
254  auto goalTracker = this->GetGoalTracker();
255  const std::vector<size_t> unreachedGoals = goalTracker->UnreachedGoalIndexes();
256  auto task = this->GetTask();
257  const size_t numGoals = task->GetNumGoals();
258 
259  // Record the path length history when this function exits.
260  nonstd::call_on_destruct trackHistory([this](){
261  this->GetStatClass()->AddToHistory("pathlength", this->GetPath()->Length());
262  });
263 
264  if(goalTracker->GetStartVIDs().empty()) {
265  if(this->m_debug)
266  std::cout << "No start VIDs, query cannot succeed."
267  << std::endl;
268  return false;
269  }
270 
271  // If no goals remain, then this must be a refinement step (as in optimal
272  // planning). In this case or the roadmap has changed, reinitialize and
273  // rebuild the whole path.
274  auto r = this->GetRoadmap();
275  if(unreachedGoals.empty() or r != m_roadmap or task != m_task)
276  Reset(r);
277 
278  if(this->m_debug)
279  std::cout << "Querying roadmap for a path satisfying task '"
280  << task->GetLabel()
281  << "', " << unreachedGoals.size() << " / " << numGoals
282  << " goals not reached."
283  << "\n\tTrying to connect goal: " << m_goalIndex
284  << "\n\tUnreached goals: " << unreachedGoals
285  << std::endl;
286 
287  // Search for a sequential path through each task constraint in order.
288  auto path = this->GetPath();
289  for(; m_goalIndex < numGoals; ++m_goalIndex) {
290  // If this goal constraint is unreached, the query cannot succeed.
291  auto iter = std::find(unreachedGoals.begin(), unreachedGoals.end(),
292  m_goalIndex);
293  const bool unreached = iter != unreachedGoals.end();
294  if(unreached) {
295  if(this->m_debug)
296  std::cout << "\tGoal " << m_goalIndex << " has no satisfying VIDs."
297  << std::endl;
298  return false;
299  }
300 
301  // Get the start VID for this subquery.
302  const VID start = path->Empty() ? *goalTracker->GetStartVIDs().begin()
303  : path->VIDs().back();
304 
305  // Get the goal VIDs for this subquery.
306  const VIDSet& goals = goalTracker->GetGoalVIDs(m_goalIndex);
307  if(goals.empty())
308  throw RunTimeException(WHERE) << "No VIDs located for reached goal "
309  << m_goalIndex << ".";
310 
311  if(this->m_debug)
312  std::cout << "\tEvaluating sub-query from " << start << " to " << goals
313  << "." << std::endl;
314 
315  // Warn users if multiple goals are found.
316  if(goals.size() > 1 and numGoals > 1)
317  std::cerr << "\tWarning: subquery has " << goals.size() << " possible VIDs "
318  << "for goal " << m_goalIndex << "/" << numGoals
319  << ". The algorithm will try its best but isn't complete for "
320  << "this case." << std::endl;
321 
322  // Perform this subquery. If it fails, there is no path.
323  const bool success = PerformSubQuery(start, goals);
324  if(!success)
325  return false;
326  }
327 
328  this->GetStatClass()->SetStat(this->GetNameAndLabel() + "::FoundPath", 1);
329 
330  if(this->m_debug)
331  std::cout << "\tConnected all goals!" << std::endl;
332 
333  return true;
334 }

340 QueryMethod<MPTraits>::
341 GeneratePath(const VID _start, const VIDSet& _goals) {
342  auto stats = this->GetStatClass();
343  MethodTimer mt(stats, "QueryMethod::GeneratePath");
344 
345  if(this->m_debug)
346  std::cout << "Generating path from " << _start << " to " << _goals << "."
347  << std::endl;
348 
349  // Check for trivial path.
350  if(_goals.count(_start) and m_startTime >= m_endTime)
351  return {_start};
352 
353  // Check for impossibility with CC information (if available).
354  auto ccTracker = m_roadmap->GetCCTracker();
355  if(ccTracker) {
356  const bool impossible = std::none_of(_goals.begin(), _goals.end(),
357  [ccTracker, _start](const VID _vid) {
358  return ccTracker->InSameCC(_start, _vid);
359  }
360  );
361 
362  if(impossible) {
363  if(this->m_debug)
364  std::cout << "\tStart and goal in different CCs, no path exists"
365  << std::endl;
366  return {};
367  }
368  }
369 
370  stats->IncStat("Graph Search");
371 
372  if(m_twoVariable)
373  return TwoVariableQuery(_start,_goals);
374 
375  // Set up the termination criterion to quit early if we find a goal node.
376  SSSPTerminationCriterion<RoadmapType> termination(
377  [_goals](typename RoadmapType::vertex_iterator& _vi,
378  const SSSPOutput<RoadmapType>& _sssp) {
379  return _goals.count(_vi->descriptor()) ? SSSPTermination::EndSearch
380  : SSSPTermination::Continue;
381  }
382  );
383 
384  // Set up the path weight function depending on whether we have any dynamic
385  // obstacles.
386  SSSPPathWeightFunction<RoadmapType> weight;
387  if(m_weightFunction) {
388  weight = m_weightFunction;
389  }
390  else if(!this->GetMPProblem()->GetDynamicObstacles().empty()) {
391  weight = [this](typename RoadmapType::adj_edge_iterator& _ei,
392  const double _sourceDistance,
393  const double _targetDistance) {
394  return this->DynamicPathWeight(_ei, _sourceDistance, _targetDistance);
395  };
396  }
397  else {
398  weight = [this](typename RoadmapType::adj_edge_iterator& _ei,
399  const double _sourceDistance,
400  const double _targetDistance) {
401  return this->StaticPathWeight(_ei, _sourceDistance, _targetDistance);
402  };
403  }
404 
405  // Run dijkstra's algorithm to find the path, if it exists.
406  const SSSPOutput<RoadmapType> sssp = DijkstraSSSP(m_roadmap, {_start}, weight,
407  termination);
408 
409  // Find the last discovered node, which should be a goal if there is a valid
410  // path.
411  const VID last = sssp.ordering.back();
412  if(!_goals.count(last)) {
413  if(this->m_debug)
414  std::cout << "\tFailed: could not find a path." << std::endl;
415  return {};
416  }
417 
418  // Extract the path.
419  std::vector<VID> path;
420  path.push_back(last);
421 
422  VID current = last;
423  do {
424  current = sssp.parent.at(current);
425  path.push_back(current);
426  } while(current != _start);
427  std::reverse(path.begin(), path.end());
428 
429  if(this->m_debug)
430  std::cout << "\tSuccess: reached goal node " << last << " with path cost "
431  << sssp.distance.at(last) << "."
432  << std::endl;
433 
434  return path;
435 }

532 double
533 QueryMethod<MPTraits>::
534 DynamicPathWeight(typename RoadmapType::adj_edge_iterator& _ei,
535  const double _sourceDistance, const double _targetDistance) const {
536  // Compute the new 'distance', which is the number of timesteps at which
537  // the robot would reach the target node.
538  const double edgeWeight = _ei->property().GetTimeSteps(),
539  newDistance = _sourceDistance + edgeWeight;
540 
541  // If this edge isn't better than the previous, we won't use it regardless
542  // and can return without checking the dynamic obstacles.
543  if(newDistance >= _targetDistance) {
544  if(this->m_debug)
545  std::cout << "Breaking because the path is not optimal."
546  << std::endl;
547  return newDistance;
548  }
549 
550  auto siTool = this->GetMPTools()->GetSafeIntervalTool(m_safeIntervalLabel);
551 
552  // Ensure that the target vertex is contained within a SafeInterval when
553  // arriving.
554  const auto& target = m_roadmap->GetVertex(_ei->target());
555  auto vertexIntervals = siTool->ComputeIntervals(target);
556  if(!(siTool->ContainsTimestep(vertexIntervals, newDistance))) {
557  if(this->m_debug)
558  std::cout << "Breaking because the target vertex is dynamically invalid."
559  << "\n\tvertexIntervals: " << vertexIntervals
560  << std::endl;
561  return std::numeric_limits<double>::infinity();
562  }
563 
564  // Ensure that the edge is contained within a SafeInterval if leaving now.
565  auto edgeIntervals = siTool->ComputeIntervals(_ei->property(), _ei->source(),
566  _ei->target(), m_roadmap);
567  if(!(siTool->ContainsTimestep(edgeIntervals, _sourceDistance))){
568  if(this->m_debug)
569  std::cout << "Breaking because the edge is dynamically invalid."
570  << std::endl;
571  return std::numeric_limits<double>::infinity();
572  }
573 
574  // If we're still here, the edge is OK.
575  return newDistance;
576 }

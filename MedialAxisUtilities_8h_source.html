295 MedialAxisUtility<MPTraits>::
296 PushCfgToMedialAxisMidpointRule(CfgType& _cfg, const Boundary* const _b) {
297  //This function is using the premise of taking the midpoint of
298  // 2 cfgs in C-Space and calling it the MA. Basically the witness point and
299  // the normal to push the cfg along are found as usual, but this normal is
300  // simply pushed along until another collision is found, then the midpoint
301  // between the two is called the MA cfg.
302  //This method means that the witness only needs to be found at the beginning,
303  // then the cfg is pushed (checking collisions still) along that line. Only
304  // doing the witness/ray shooting a single time is a HUGE benefit.
305  //When using Exact clearance/witness finding, the iterations stop when the
306  // witness changes. We can only do this in exact since the approximate witness
307  // finding is extremely noise-prone and slow.
308  string callee = this->GetNameAndLabel() + "::PushCfgToMedialAxisMidpointRule";
309  if(this->m_debug)
310  cout << callee << endl << "Cfg: " << _cfg << endl;
311 
312  auto vcm = this->GetValidityChecker(this->m_vcLabel);
313 
314  CDInfo tmpInfo;
315  CDInfo firstWitnessInfo;
316  tmpInfo.ResetVars(true);//true so it returns all info.
317 
318  // Should already be in free space
319  if(!vcm->IsValid(_cfg, tmpInfo, callee)) {
320  if(this->m_debug)
321  std::cout << callee <<": Returning false due to already invalid"
322  << std::endl;
323  return false;
324  }
325 
326  //Get the first witness:
327  CfgType firstWitnessCfg(_cfg.GetRobot());
328  if(!this->CollisionInfo(_cfg, firstWitnessCfg, _b, tmpInfo, true)) {
329  if(this->m_debug)
330  std::cout << callee << ": Returning false from not being able to"
331  " find first witness" << std::endl;
332  return false;
333  }
334  firstWitnessInfo = tmpInfo;
335 
336  Vector3d normalDirection = (tmpInfo.m_robotPoint - tmpInfo.m_objectPoint).normalize();
337 
338  //Find the unit normal:
339 // CfgType unitDirectionToTickCfgAlong = _Cfg - firstWitnessCfg;
340 // unitDirectionToTickCfgAlong /= (unitDirectionToTickCfgAlong.Magnitude());
341  CfgType unitDirectionToTickCfgAlong;
342 
343  if(this->m_exactClearance)
344  unitDirectionToTickCfgAlong = CfgType(normalDirection, _cfg.GetRobot());
345  else {
346  unitDirectionToTickCfgAlong = _cfg - firstWitnessCfg;
347  unitDirectionToTickCfgAlong /= unitDirectionToTickCfgAlong.Magnitude();
348  }
349 
350  if(this->m_debug)
351  std::cout << callee << ": first normal directional cfg from witness = "
352  << unitDirectionToTickCfgAlong << std::endl
353  << "_cfg = " << _cfg << std::endl << "firstWitnessCfg = "
354  << firstWitnessCfg << std::endl;
355 
356  //Move along normal until anther witness is found.
357  //A better maxIterations might be in order, but this should be fine for now.
358  const double maSearchResolution = this->GetEnvironment()->GetPositionRes()
359  * this->m_maSearchResolutionFactor;
360  const size_t maxIterations = _b->GetMaxDist()/this->m_rayTickResolution;
361  CfgType magnitudeAndDirectionToTick = unitDirectionToTickCfgAlong
362  * maSearchResolution;
363 
364  bool inBounds = true;
365  bool valid = true;
366  bool passed = false;
367  CfgType tickedCfg = _cfg;
368  mathtool::Vector3d firstWitnessVertex = tmpInfo.m_objectPoint;
369 
370  //The MA search loop:
371  for(size_t tick = 1; tick < maxIterations; tick++) {
372  //Tick the cfg:
373  tickedCfg += magnitudeAndDirectionToTick;
374 
375  //Check if in bounds:
376  inBounds = tickedCfg.InBounds(_b);
377  if(!inBounds && !this->m_useBBX) {
378  if(this->m_debug)
379  std::cout << "Returning false from going OOB but not using BBX as "
380  "obstacle" << std::endl;
381  return false;
382  }
383 
384  if(this->m_exactClearance) {
385  //Exact, so we want all CD info, specifically the witness point.
386  // GetNearestVertexWitness will do this for us, including boundary witness
387  // information, if BBX is active as an obstacle. It also returns the
388  // IsValid() boolean for _cfg and puts all data in tmpInfo:
389  valid = this->GetNearestVertexWitness(tickedCfg, tmpInfo, _b);
390 
391  if(!valid || !inBounds) {
392  if(this->m_debug)
393  std::cout << "Returning false from going invalid when looking for "
394  "second exact witness." << std::endl
395  << "_cfg = " << _cfg << std::endl << "first witness cfg = "
396  << firstWitnessCfg << std::endl << "unit tick direction = "
397  << unitDirectionToTickCfgAlong << std::endl << "Current "
398  "ticked cfg = " << tickedCfg << std::endl << std::endl;
399  return false;
400  }
401  //For exact, we can quit as soon as we have a witness change.
402  //I found some issues initially when doing a pure equality, so check
403  // the two witness vertices are sufficiently different.
404  //static const double threshold = this->GetEnvironment()->GetPositionRes()
405  // * pow(2, _cfg->posDOF());
406 
407  // @TODO: check the witnesses are sufficiently different before it is
408  // considered valid. Currently, using double the environment resolution is
409  // a quick fix; however, we should be checking the relative difference
410  // between the cfg and its witness to check if the witnesses are
411  // sufficiently different from the first witness. It is not clear as to
412  // how to do this
413  //passed = !FuzzyVectorEquality(
414  // tmpInfo.m_objectPoint, firstWitnessVertex, threshold);
415  passed = !WitnessObstacleEquality(_cfg, firstWitnessInfo, tmpInfo);
416  }
417  else {
418  //Do the most simple validity check (not returning all info).
419  valid = vcm->IsValid(tickedCfg, callee);
420  passed = !inBounds || !valid;
421  }
422 
423  if(passed) {
424  if(this->m_debug && this->m_exactClearance)
425  std::cout << callee << ": passed on tick " << tick << " with original "
426  << "witness vertex = " << firstWitnessVertex << " and second "
427  "witness vertex = " << tmpInfo.m_objectPoint
428  << std::endl << std::endl;
429  break;// we have found the second and final witness, break out of the loop
430  }
431  }
432 
433  //check that it passed, and didn't just time out.
434  if(!passed) {
435  if(this->m_debug)
436  std::cout << callee << " Returning false as no second witness could "
437  "be found" << std::endl;
438  return false;
439  }
440 
441  //If doing exact, then the MA sample is the midpoint of the two last ticks.
442  if(this->m_exactClearance)
443  firstWitnessCfg = tickedCfg - magnitudeAndDirectionToTick;
444 
445  //If doing approx, the first witness is already set, and the last witness is
446  // simply the tickedCfg.
447 
448  //Find the midpoint of the two cfgs as the MA cfg:
449  CfgType cfgMA = (firstWitnessCfg + tickedCfg)/2.0;
450 
451  //Now we have to double check that it's valid, since it's possible that it's
452  // not, especially with increase in complexity of environment/narrow passages.
453  if(!cfgMA.InBounds(_b) || !vcm->IsValid(cfgMA, callee)) {
454  //It's either OOB or it's invalid. This pair of witness won't work for a
455  // MA sample, so return false.
456  if(this->m_debug)
457  std::cout << callee << "Returning false due to invalid midpoint cfg"
458  << std::endl;
459  return false;
460  }
461 
462  _cfg = cfgMA;
463 
464  if(this->m_debug) {
465  VDComment("Final CFG");
466  VDAddTempCfg(cfgMA, true);
467  VDClearLastTemp();
468  VDClearComments();
469  VDClearLastTemp();
470  VDClearLastTemp();
471  cout << "FINAL Cfg: " << _cfg << endl;
472  }
473 
474  return true;
475 }

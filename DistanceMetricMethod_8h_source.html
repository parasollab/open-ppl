145 DistanceMetricMethod<MPTraits>::
146 EdgeWeight(const RoadmapType* const _r, const typename RoadmapType::CEI& _edge)
147  noexcept {
148  // Get the intermediates. If there aren't any, the weight is just from source
149  // to target.
150  const auto& edge = *_edge;
151  const auto& intermediates = edge.property().GetIntermediates();
152  const typename RoadmapType::VP& source = _r->GetVertex(edge.source()),
153  & target = _r->GetVertex(edge.target());
154  if(intermediates.empty())
155  return Distance(source, target);
156 
157  // If we're still here, there are intermediates. Add up the distance through
158  // the intermediates.
159  double totalDistance = Distance(source, intermediates.front())
160  + Distance(intermediates.back(), target);
161 
162  for(size_t i = 0; i < intermediates.size() - 1; ++i)
163  totalDistance += Distance(intermediates[i], intermediates[i + 1]);
164 
165  return totalDistance;
166 }

189 ScaleCfg(double _length, CfgType& _c, const CfgType& _o) {
194 
195  _length = fabs(_length); //a distance must be positive
196  CfgType origin = _o;
197  CfgType outsideCfg = _c;
198 
199  // first find an outsite configuration with sufficient size
200  while(Distance(origin, outsideCfg) < 2 * _length)
201  for(size_t i = 0; i < outsideCfg.DOF(); ++i)
202  outsideCfg[i] *= 2.0;
203  // now, using binary search find a configuration with the approximate length
204  CfgType aboveCfg = outsideCfg;
205  CfgType belowCfg = origin;
206  CfgType currentCfg = _c;
207  while (1) {
208  for(size_t i=0; i<currentCfg.DOF(); ++i)
209  currentCfg[i] = (aboveCfg[i] + belowCfg[i]) / 2.0;
210  double magnitude = Distance(origin, currentCfg);
211  if((magnitude >= _length*0.9) && (magnitude <= _length*1.1))
212  break;
213  if(magnitude>_length)
214  aboveCfg = currentCfg;
215  else
216  belowCfg = currentCfg;
217  }
218 
219  _c = currentCfg;
220 }

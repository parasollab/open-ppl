149 FindNeighborsImpl(AbstractRoadmapType* const _r,
150  const typename AbstractRoadmapType::CfgType& _cfg,
151  const VertexSet& _candidates, OutputIterator _out) {
152  auto dm = this->GetDistanceMetric(this->m_dmLabel);
153 
154  // The neighbor to use as a fallback if m_useFallback is set.
155  Neighbor fallback;
156 
157  // Find all nodes within radius
158  std::multiset<Neighbor> inRadius;
159 
160  for(const VID vid : _candidates) {
161  // Check for connectedness.
162  if(this->m_unconnected and this->DirectEdge(_r, _cfg, vid))
163  continue;
164 
165  // Check for connection to self.
166  const auto& node = _r->GetVertex(vid);
167  if(node == _cfg)
168  continue;
169 
170  // Check distance. If it is infinite, this is not connectable.
171  const double distance = dm->Distance(_cfg, node);
172  if(std::isinf(distance))
173  continue;
174 
175  // If within radius, add to list. If not, check for better fallback node.
176  if(distance <= this->m_radius)
177  inRadius.emplace(vid, distance);
178  else if(m_useFallback and distance < fallback.distance) {
179  fallback.distance = distance;
180  fallback.target = vid;
181  }
182  }
183 
184  // Return fallback if no nodes were found within the radius.
185  if(m_useFallback and inRadius.empty())
186  inRadius.insert(fallback);
187 
188  std::copy(inRadius.begin(), inRadius.end(), _out);
189 }

527 template <typename MPTraits>
528 const Vector3d
529 DynamicRegionRRT<MPTraits>::
530 GetVelocityBias(SamplingRegion* _region) {
531  // Get the region data.
532  // const auto& regionData = m_regionData.at(region);
533  const size_t index = _region->edgeIndex;
534 
535  // Find the skeleton edge path the region is traversing.
536  auto reit = _region->edgeIterator;
537  const auto& path = reit->property();
538 
539  // Helper to make the biasing direction and print debug info.
540  auto makeBias = [&](const Vector3d& _start, const Vector3d& _end) {
541  if(this->m_debug)
542  std::cout << "Computed velocity bias: " << (_end - _start).normalize()
543  << "\n\tStart: " << _start
544  << "\n\tEnd: " << _end
545  << std::endl;
546  return (_end - _start).normalize();
547  };
548 
549  // If there is at least one valid path point after the current path index,
550  // then return the direction to the next point.
551  if(index < path.size() - 1) {
552  if(this->m_debug)
553  std::cout << "Biasing velocity along next path step"
554  << "\n\tPath index: " << index
555  << "\n\tPath size: " << path.size()
556  << std::endl;
557  return makeBias(path[index], path[index + 1]);
558  }
559 
560  // Otherwise, the region has reached a skeleton vertex.
561  WorkspaceSkeleton::VD targetVD = reit->target();
562  auto vertex = m_skeleton.FindVertex(targetVD);
563 
564  // If the vertex has no outgoing edges, this is the end of the skeleton. In
565  // that case, use the previous biasing direction. All paths have at least two
566  // points so this is safe.
567  if(vertex->size() == 0) {
568  if(this->m_debug)
569  std::cout << "Biasing velocity along previous path step"
570  << "\n\tPath index: " << index
571  << "\n\tPath size: " << path.size()
572  << std::endl;
573  return makeBias(path[index - 1], path[index]);
574  }
575 
576  // Otherwise, randomly select an outgoing and use it's next point.
577  auto eit = vertex->begin();
578  const size_t nextEdgeIndex = LRand() % vertex->size();
579  std::advance(eit, nextEdgeIndex);
580  if(this->m_debug)
581  std::cout << "Biasing velocity along next edge (index " << nextEdgeIndex
582  << ")\n\tPath index: " << index
583  << "\n\tPath size: " << path.size()
584  << "\n\tNext edge path size: " << eit->property().size()
585  << std::endl;
586  return makeBias(path[index], eit->property()[1]);
587 }

590 template <typename MPTraits>
591 bool
592 DynamicRegionRRT<MPTraits>::
593 IsTouching(const Cfg& _cfg, SamplingRegion& _region) {
594  // Compute the penetration distance required. We want the robot's bounding
595  // sphere to penetrate the region by the fraction m_penetrationThreshold.
596  const double robotRadius = _cfg.GetMultiBody()->GetBoundingSphereRadius(),
597  threshold = 2 * robotRadius * m_penetrationFactor;
598 
599  // Get the region boundary.
600  const auto center = _region.GetCenter();
601  auto boundary = MakeBoundary(center);
602 
603  // Compute the penetration distance (maximally enclosed bounding diameter).
604  const Point3d robotCenter = _cfg.GetPoint();
605  const double penetration = boundary.GetClearance(robotCenter) + robotRadius;
606 
607  // The configuration is touching if the penetration exceeds the threshold.
608  const bool touching = penetration >= threshold;
609 
610  if(this->m_debug)
611  std::cout << "\t Touch test: " << (touching ? "passed" : "failed")
612  << "\n\t Bounding sphere: " << robotCenter << " ; " << robotRadius
613  << "\n\t Region: " << _region.GetCenter() << " ; "
614  << m_regionRadius
615  << "\n\t Bounding sphere penetrates by "
616  << std::setprecision(4)
617  << penetration << (touching ? " >= " : " < ") << threshold
618  << " units."
619  << std::endl;
620 
621  return touching;
622 }

649 DynamicRegionRRT<MPTraits>::
650 BuildSkeleton() {
651  auto stats = this->GetStatClass();
652  MethodTimer mt(stats, this->GetNameAndLabel() + "::BuildSkeleton");
653 
654  // Determine if we need a 2d or 3d skeleton.
655  auto env = this->GetEnvironment();
656  auto robot = this->GetTask()->GetRobot();
657  const bool threeD = robot->GetMultiBody()->GetBaseType() ==
658  Body::Type::Volumetric;
659 
660  if(threeD) {
661  if(m_skeletonType == "mcs") {
662  if(this->m_debug)
663  std::cout << "Building a Mean Curvature skeleton." << std::endl;
664  MeanCurvatureSkeleton3D mcs;
665  mcs.SetEnvironment(this->GetEnvironment());
666  mcs.BuildSkeleton();
667 
668  // Create the workspace skeleton.
669  auto sk = mcs.GetSkeleton();
670  m_originalSkeleton = sk.first;
671  m_originalSkeleton.DoubleEdges();
672  }
673  else if(m_skeletonType == "reeb") {
674  // Create a workspace skeleton using a reeb graph.
675  if(this->m_debug)
676  std::cout << "Building a Reeb Graph skeleton." << std::endl;
677  auto decomposition = this->GetMPTools()->GetDecomposition(
678  m_decompositionLabel);
679  ReebGraphConstruction reeb;
680  reeb.Construct(decomposition);
681 
682  // Create the workspace skeleton.
683  m_originalSkeleton = reeb.GetSkeleton();
684  m_originalSkeleton.DoubleEdges();
685  }
686  else
687  throw ParseException(WHERE) << "Unrecognized skeleton type '"
688  << m_skeletonType << "', options for 3d "
689  << "problems are {mcs, reeb}.";
690  }
691  else {
692  // Collect the obstacles we want to consider (all in this case).
693  std::vector<GMSPolyhedron> polyhedra;
694  for(size_t i = 0; i < env->NumObstacles(); ++i) {
695  MultiBody* const obstacle = env->GetObstacle(i);
696  for(size_t j = 0; j < obstacle->GetNumBodies(); ++j)
697  polyhedra.emplace_back(obstacle->GetBody(j)->GetWorldPolyhedron());
698  }
699 
700  // Build a skeleton from a 2D medial axis.
701  if(this->m_debug)
702  std::cout << "Build a skeleton from a 2D medial axis." << endl;
703  MedialAxis2D ma(polyhedra, env->GetBoundary());
704  ma.BuildMedialAxis();
705  m_originalSkeleton = get<0>(ma.GetSkeleton(1)); // 1 for free space.
706  }
707 
708  if(this->m_debug)
709  std::cout << "Direct skeleton" << endl;
710  DirectSkeleton();
711 }

714 template <typename MPTraits>
715 void
716 DynamicRegionRRT<MPTraits>::
717 DirectSkeleton() {
718  // Only support single-goal tasks; this is inherent to the method. The problem
719  // is solvable but hasn't been solved yet.
720  const auto& goalConstraints = this->GetTask()->GetGoalConstraints();
721  if(goalConstraints.size() != 1)
722  throw RunTimeException(WHERE) << "Only supports single-goal tasks. "
723  << "Multi-step tasks will need new skeletons "
724  << "for each sub-component.";
725 
726  // Find the workspace points which are nearest to the start and goal.
727  auto g = this->GetRoadmap();
728  auto goalTracker = this->GetGoalTracker();
729  const auto& startVIDs = goalTracker->GetStartVIDs();
730  const auto& goalVIDs = goalTracker->GetGoalVIDs(0);
731  Point3d start, goal;
732  if(startVIDs.size() == 1) {
733  const VID startVID = *startVIDs.begin();
734  start = g->GetVertex(startVID).GetPoint();
735  }
736  else {
737  // Probably we can just take the center of the start constraint boundary if
738  // applicable, although we have no cases requiring that right now.
739  throw RunTimeException(WHERE) << "Exactly one start VID is required, but "
740  << startVIDs.size() << " were found.";
741  }
742 
743  if(goalVIDs.size() == 1) {
744  const VID goalVID = *goalVIDs.begin();
745  goal = g->GetVertex(goalVID).GetPoint();
746  }
747  else {
748  // Check for a goal boundary. We already checked that there is one goal
749  // constraint, so it is safe to assume it exists here.
750  const Boundary* const boundary = goalConstraints[0]->GetBoundary();
751  if(!boundary)
752  throw RunTimeException(WHERE) << "Exactly one goal VID is required, but "
753  << goalVIDs.size() << " were found and no "
754  << "constraint boundary was available.";
755 
756  // Try to sample a configuration in the boundary.
757  auto sampler = this->GetSampler(this->m_samplerLabel);
758  const size_t count = 1,
759  attempts = 100;
760  std::vector<CfgType> samples;
761  sampler->Sample(count, attempts, boundary, std::back_inserter(samples));
762 
763  // If we couldn't generate a configuration here, the goal boundary isn't
764  // realistic.
765  if(samples.empty())
766  throw RunTimeException(WHERE) << "Could not generate a sample within the "
767  << "goal boundary " << *boundary
768  << " after " << attempts << " attempts.";
769 
770  // We got a sample, take its point as the center point.
771  goal = samples.front().GetPoint();
772  }
773 
774  // If there is a new start and goal pair, redirect the skeleton
775  std::pair<Point3d, Point3d> currentQuery{start, goal};
776  if(currentQuery == m_queryPair)
777  return;
778

844 template <typename MPTraits>
845 std::vector<double>
846 DynamicRegionRRT<MPTraits>::
847 ComputeProbabilities() {
848 
849  // Sum all weights of all current regions.
850  double totalWeight = 0.;
851  for(auto r : m_regions) {
852  totalWeight += r.GetWeight();
853  }
854 
855  // Compute the probabilities for the current regions.
856  std::vector<double> probabilities;
857  probabilities.reserve(m_regions.size() + 1);
858 
859  const double explore = m_explore / (m_regions.size() + 1);
860 
861  for(auto r : m_regions) {
862  const double exploit = (1 - m_explore) * r.GetWeight() / totalWeight;
863 
864  probabilities.emplace_back(exploit + explore);
865  }
866 
867  // Get the probability for the whole environment.
868  probabilities.emplace_back(explore);
869 
870  return probabilities;
871 }

949 template <typename MPTraits>
950 void
951 DynamicRegionRRT<MPTraits>::
952 AdvanceRegions(const Cfg& _cfg) {
953  if(this->m_debug)
954  std::cout << "Checking " << m_regions.size()
955  << " regions for contact with new configuration "
956  << _cfg.PrettyPrint() << "."
957  << std::endl;
958 
959  // Keep track of any newly reached vertices to spawn regions on their outbound
960  // edges.
961  std::queue<WorkspaceSkeleton::VD> newlyReachedVertices;
962 
963  // Iterate through all existing regions to see which should be advanced.
964  for(auto iter = m_regions.begin(); iter != m_regions.end(); ) {
965 
966  // Advance this region until the robot at _cfg is no longer touching it.
967  if(!AdvanceRegionToCompletion(_cfg, *iter)) {
968  ++iter;
969  continue;
970  }
971 
972  // We have reached the end of this region's edge. Delete it and save the
973  // target vertex. to spawn new regions.
974  auto eit = iter->edgeIterator;
975  const auto target = eit->target();
976  if(!m_visited[target])
977  newlyReachedVertices.push(target);
978  iter = m_regions.erase(iter);
979  }
980 
981  // Create new regions for each newly reached vertex.
982  while(!newlyReachedVertices.empty()) {
983  // Pop the next vertex off the queue.
984  WorkspaceSkeleton::VD targetVD = newlyReachedVertices.front();
985  WorkspaceSkeleton::vertex_iterator target = m_skeleton.FindVertex(targetVD);
986  newlyReachedVertices.pop();
987 
988  // Create regions at this vertex.
989  std::vector<SamplingRegion*> newRegions = CreateRegions(target);
990 
991  // Advance each new region.
992  for(auto region : newRegions) {
993  // Advance this region until the robot at _cfg is no longer touching it.
994  if(!AdvanceRegionToCompletion(_cfg, *region))
995  continue;
996 
997  // We have reached the end of this region's edge. Delete it and save the
998  // target vertex. to spawn new regions.
999  auto iter = std::find(m_regions.begin(), m_regions.end(), *region);
1000  auto eit = region->edgeIterator;
1001  newlyReachedVertices.push(eit->target());
1002  m_regions.erase(iter);
1003  }
1004  }
1005 }

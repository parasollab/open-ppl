635 MPToolsType<MPTraits>::
636 GetDecomposition(const std::string& _label) {
637  // Initialize the decomposition if not already.
638  typename decltype(m_decompositions)::iterator iter;
639  try {
640  iter = m_decompositions.find(_label);
641  }
642  catch(const std::out_of_range&) {
643  throw RunTimeException(WHERE) << "Requested decomposition '" << _label
644  << "' does not exist.";

646 
647  if(iter->second == nullptr) {
648  MethodTimer mt(m_library->GetStatClass(), "TetGenDecomposition::" + _label);
649  iter->second = m_tetgens[_label](m_library->GetMPProblem()->GetEnvironment());
650  }

741 template <typename MPTraits>
742 template <template <typename> class Utility>
743 void
744 MPToolsType<MPTraits>::
745 SetUtility(const std::string& _label, Utility<MPTraits>* _utility,
746  LabelMap<Utility>& _map) const {
747  // Set the library pointer.
748  _utility->SetMPLibrary(m_library);
749 
750  // Check if this label is already in use.
751  auto iter = _map.find(_label);
752  const bool alreadyExists = iter != _map.end();
753 
754  // If the label already exists, we need to release the previous utility first.
755  if(alreadyExists) {
756  delete iter->second;
757  iter->second = _utility;
758  }
759  else
760  _map.insert({_label, _utility});
761 }

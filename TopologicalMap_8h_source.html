522 template <typename MPTraits>
523 const WorkspaceRegion*
524 TopologicalMap<MPTraits>::
525 LocateRegion(const Point3d& _point) const {
526  MethodTimer mt(this->GetStatClass(),
527  this->GetNameAndLabel() + "::LocateRegion");
528 
529  // Find the grid cell that contains the new configuration's reference point.
530  const size_t cell = m_grid->LocateCell(_point);
531 
532  // Find the correct region out of the candidates.
533  const auto& candidateRegions = m_cellToRegions[cell];
534 
535  const WorkspaceRegion* r = nullptr;
536  for(const WorkspaceRegion* region : candidateRegions) {
537  if(region->GetBoundary()->InBoundary(_point)) {
538  r = region;
539  break;
540  }
541  }
542 
543  if(this->m_debug) {
544  // Use pqp to check if _point lies within an obstacle.
545  using CDType = CollisionDetectionValidity<MPTraits>;
546  auto vc = static_cast<CDType*>(this->GetValidityChecker(m_pqpLabel));
547  const bool inObstacle = vc->IsInsideObstacle(_point);
548 
549  // Check for boundary containment.
550  auto env = this->GetEnvironment()->GetBoundary();
551  const bool inBoundary = env->InBoundary(_point);
552 
553  std::cout << "TopologicalMap::LocateRegion"
554  << "\n\tPoint " << _point << "is " << (inObstacle ? "" : "not ")
555  << "inside an obstacle."
556  << "\n\tPoint " << _point << "is " << (inBoundary ? "" : "not ")
557  << "inside the env boundary."
558  << "\n\tCell: " << cell
559  << " with center " << m_grid->CellCenter(cell)
560  << ", width " << m_grid->CellLength()
561  << "\n\tCandidate regions: " << candidateRegions.size()
562  << "\n\tContaining region: " << r
563  << std::endl;
564 
565  // If the point isn't in an obstacle and we have no region, then something
566  // went wrong with the cell to region map.
567  // Build an obj mesh of the candidate regions, and a small box for the
568  // point. Merge these into a single obj and print to file for inspection.
569  if(!inObstacle and !r) {
570  glutils::triangulated_model t;
571 
572  // Add the env boundary so that we know roughly where in the environment
573  // the problem occured.
574  auto b = glutils::triangulated_model::make_box(env->GetRange(0).Length(),
575  env->GetRange(1).Length(), env->GetRange(2).Length());
576  {
577  const std::vector<double>& center = env->GetCenter();
578  b.translate(ToGLUtils(Point3d(center[0], center[1], center[2])));
579  t.add_model(b);
580  }
581 
582  // Add the candidate regions, and find their max bounding sphere size.
583  double maxRadius = 0.;
584  for(const WorkspaceRegion* region : candidateRegions) {
585  // Add the facets.
586  for(const WorkspaceRegion::Facet facet : region->GetFacets()) {
587  t.add_facet(t.add_point(ToGLUtils(facet.GetPoint(0))),
588  t.add_point(ToGLUtils(facet.GetPoint(1))),
589  t.add_point(ToGLUtils(facet.GetPoint(2))));
590  }
591 
592  // Find the center.
593  const Point3d c = region->FindCenter();
594  // Find max distance from center.
595  for(const Point3d& p : region->GetPoints())
596  maxRadius = std::max(maxRadius, (p-c).norm());
597  }
598 
599  // Add the cell.
600  auto c = glutils::triangulated_model::make_box(m_grid->CellLength(),
601  m_grid->CellLength(), m_grid->CellLength());
602  c.translate(ToGLUtils(m_grid->CellCenter(cell)));
603  t.add_model(c);
604 
605  // Add the point as a small sphere of 1/20 the radius.
606  auto p = glutils::triangulated_model::make_sphere(maxRadius / 20., 4);
607  p.translate(ToGLUtils(_point));
608  t.add_model(p);
609 
610  // Output the model to an obj file.
611  {
612  glutils::obj_file file("tm-error.obj");
613  file << t;
614  }
615 
616  throw RunTimeException(WHERE) << "Freespace point " << _point
617  << " is not in any region.";
618  }
619  }
620 
621  return r;
622 }

638 TopologicalMap<MPTraits>::
639 LocateNearestRegion(const Point3d& _p) const {
640  MethodTimer mt(this->GetStatClass(),
641  this->GetNameAndLabel() + "::LocateNearestRegion");
642 
643  // Put the point robot here.
644  auto pointRobot = this->GetMPProblem()->GetRobot("point");
645  CfgType temp(_p, pointRobot);
646 
647  // Use PQPSolid to get the CD info. If the check is valid, this was already in
648  // free space.
649  CDInfo cdInfo(true);
650  auto vc = this->GetValidityChecker(m_pqpLabel);
651  if(vc->IsValid(temp, cdInfo, this->GetNameAndLabel() + "::LocateNearestRegion"))

655  const Vector3d delta = cdInfo.m_objectPoint - cdInfo.m_robotPoint;
656  auto env = this->GetEnvironment();

660  return LocateRegion(cdInfo.m_objectPoint + delta.scale(env->GetPositionRes()));

733 template <typename MPTraits>
734 void
735 TopologicalMap<MPTraits>::
736 EnsureMap(RoadmapType* const _r) {
737  // If we already have maps for _r, do nothing.
738  if(m_regionToVIDs.count(_r))
739  return;
740 
741  // Create forward and inverse maps.
742  auto& forward = m_regionToVIDs[_r];
743  auto& inverse = m_vidToNeighborhood[_r];
744 
745  // Set the size of the mappings.
746  auto mb = _r->GetRobot()->GetMultiBody();
747  forward.resize(mb->GetNumBodies());
748  inverse.reserve(_r->Size());
749 
750  // Install roadmap hooks.
751  _r->InstallHook(RoadmapType::HookType::AddVertex, this->GetNameAndLabel(),
752  [this, _r](const VI _vi){this->MapCfg(_r, _vi);});
753  _r->InstallHook(RoadmapType::HookType::DeleteVertex, this->GetNameAndLabel(),
754  [this, _r](const VI _vi){this->UnmapCfg(_r, _vi);});
755 
756  // If the graph has existing vertices, map them now.
757  for(auto iter = _r->begin(); iter != _r->end(); ++iter)
758  MapCfg(_r, iter);
759 }

784 template<typename MPTraits>
785 void
786 TopologicalMap<MPTraits>::
787 UnmapCfg(RoadmapType* const _r, const VI _vertex) {
788  MethodTimer mt(this->GetStatClass(), this->GetNameAndLabel() + "::UnmapCfg");
789 
790  const VID vid = _vertex->descriptor();
791 
792  // Assert that this vertex was mapped to a neighborhood.
793  auto& inverse = m_vidToNeighborhood[_r];
794  auto neighborhoodIter = inverse.find(vid);
795  if(neighborhoodIter == inverse.end())
796  throw RunTimeException(WHERE) << "Tried to unmap vertex " << vid
797  << ", but didn't find a neighborhood for it.";
798 
799  // Remove the mapping from each neighborhood region to this VID.
800  const auto& neighborhood = neighborhoodIter->second;
801  auto& forward = m_regionToVIDs[_r];
802  for(size_t i = 0; i < neighborhood.size(); ++i) {
803  const auto region = neighborhood[i];
804 
805  // Assert that the region was mapped to the vertex.
806  auto& regionMap = forward[i][region];
807  auto vidIter = std::find(regionMap.begin(), regionMap.end(), vid);
808 
809  if(vidIter == regionMap.end())
810  throw RunTimeException(WHERE) << "Tried to unmap vertex " << vid
811  << ", but it was missing from it's region "
812  << "map.";
813 
814  // Remove this vertex from the map.
815  regionMap.erase(vidIter);
816  }
817 
818  inverse.erase(neighborhoodIter);
819 }

866 template <typename MPTraits>
867 double
868 TopologicalMap<MPTraits>::
869 ApproximateMinimumInnerDistance(const WorkspaceRegion* const _source,
870  const WorkspaceRegion* const _target) {
871  MethodTimer mt(this->GetStatClass(),
872  this->GetNameAndLabel() + "::ApproximateMinimumInnerDistance");
873 
874  // First check for the trivial case.
875  if(_source == _target)
876  return 0;
877 
878  // Next check if we've alreay cached inner distances from this source.
879  auto sourceIter = m_innerDistanceMap.find(_source);
880  const bool sourceCached = sourceIter != m_innerDistanceMap.end();
881  if(!sourceCached)
882  return std::numeric_limits<double>::infinity();
883 
884  // Next look for the target entry in the source map.
885  const auto& sourceMap = sourceIter->second;
886  auto targetIter = sourceMap.find(_target);
887  const bool targetCached = targetIter != sourceMap.end();
888  return targetCached
889  ? targetIter->second
890  : std::numeric_limits<double>::infinity();
891 }

894 template <typename MPTraits>
895 const typename TopologicalMap<MPTraits>::DistanceMap&
896 TopologicalMap<MPTraits>::
897 ComputeApproximateMinimumInnerDistances(const WorkspaceRegion* const _source,
898  const double _radius) {
903 
904  DistanceMap* distanceMap{nullptr};
905 
906  // Check if we've already computed for this source.
907  {
908  auto iter = m_innerDistanceMap.find(_source);
909  const bool alreadyComputed = iter != m_innerDistanceMap.end();
910 
911  // If we already computed this cell, don't do it again unless the radius is
912  // bigger.
913  if(alreadyComputed) {
914  distanceMap = &iter->second;
915  const double lastRadius = distanceMap->at(nullptr);
916  if(lastRadius >= _radius)
917  return *distanceMap;
918  }
919 
920  // We haven't computed this distance map to the necessary radius. (Re)compute
921  // with this radius.
922  distanceMap = &m_innerDistanceMap[_source];
923  (*distanceMap)[nullptr] = _radius;
924  }
925 
926  // Define a function for updating the minimum inner distance to the regions
927  // touching a given cell.
928  auto updateInnerDistance = [_source, this, distanceMap](const size_t _cell,
929  const double _distance)
930  {
931  // Find all regions touching the cell.
932  const auto& regions = this->m_cellToRegions[_cell];
933 
934  // Set the distance if it isn't already set.
935  for(const WorkspaceRegion* const region : regions) {
936  if(distanceMap->count(region))
937  continue;
938  distanceMap->emplace(region, _distance);
939  }
940  };

943  struct element {
944  size_t cell;
945  double distance;
946 
947  element(const size_t _target, const double _distance)
948  : cell(_target), distance(_distance) {}
949 
951  bool operator>(const element& _other) const noexcept {
952  return distance > _other.distance;
953  }
954  };
955 
956  // Set up a best-first search through the grid overlay to estimate the minimum
957  // inner distance.
958  std::unordered_set<size_t> visited;
959  std::unordered_map<size_t, double> distance;
960  std::unordered_map<size_t, size_t> parent;
961  std::priority_queue<element,
962  std::vector<element>,
963  std::greater<element>> pq;
964 
965  // Find the source and target cells.
966  const Boundary* const sourceBoundary = _source->GetBoundary();
967  const std::unordered_set<size_t> sourceCells = m_grid->LocateCells(
968  sourceBoundary, GridOverlay::CellSet::Closure);
969 
970  // Initialize visited and distance maps.
971  {
972  // Mark all source cells as visited and distance 0.
973  for(const size_t cell : sourceCells) {
974  visited.insert(cell);
975  distance[cell] = 0;
976  updateInnerDistance(cell, 0);
977  }
978 
979  // Find all neighbors of the source cells.
980  std::unordered_set<size_t> allNeighbors;
981  for(const size_t cell : sourceCells) {
982  std::unordered_set<size_t> neighbors = m_grid->LocateAllNeighbors(cell);
983  allNeighbors.insert(neighbors.begin(), neighbors.end());
984  }
985 
986  // Put the neighbors in the queue as visited @ distance 0.
987  for(const size_t neighbor : allNeighbors) {
988  // Skip visited cells.
989  if(visited.count(neighbor))
990  continue;
991  // Skip boundary cells.
992  if(m_boundaryCells.count(neighbor))
993  continue;
994 
995  //updateInnerDistance(neighbor, 0);
996  //visited.insert(neighbor);
997  distance[neighbor] = 0;
998  parent[neighbor] = neighbor;
999  pq.emplace(neighbor, 0);
1000  }
1001  }
1002 
1003  // Define a relax edge function.
1004  const double cellLength = this->m_grid->CellLength();
1005  auto relax = [&distance, &pq, &cellLength, &parent, this](
1006  const size_t _source, const size_t _target)
1007  {
1008  // If the target cell is a boundary cell, quit.
1009  if(m_boundaryCells.count(_target))
1010  return;
1011 
1012  // Compute the new distance.
1013  const double sourceDistance = distance[_source],
1014  targetDistance = distance.count(_target)
1015  ? distance[_target]
1016  : std::numeric_limits<double>::infinity(),
1017  newDistance = sourceDistance + cellLength;
1018 
1019  // If the new distance isn't better, quit.
1020  if(newDistance >= targetDistance)
1021  return;
1022 
1023  // Otherwise, update target distance and add the target to the queue.
1024  distance[_target] = newDistance;
1025  parent[_target] = _source;
1026  pq.emplace(_target, newDistance);
1027  };
1028 
1029  // Run Dijkstra's algorithm until discovering a cell with min distance >
1030  // _radius.
1031  while(!pq.empty()) {
1032  // Get the next element.
1033  const element current = pq.top();
1034  pq.pop();
1035 
1036  // If we are done with this node, the element is stale. Discard.
1037  if(visited.count(current.cell))
1038  continue;
1039  visited.insert(current.cell);
1040 
1041  // Check for early termination.
1042  const bool stop = current.distance > _radius;
1043 #if 0
1044  // This part is for debugging the search over the grid.

1052  break;
1053 
1054  // Update the inner distance map for all regions touched by this cell.
1055  updateInnerDistance(current.cell, current.distance);
1056 
1057  // Relax each outgoing edge.
1058  const std::unordered_set<size_t> neighbors = m_grid->LocateAllNeighbors(
1059  current.cell);
1060  for(const size_t n : neighbors)
1061  if(!visited.count(n))
1062  relax(current.cell, n);
1063  }
1064

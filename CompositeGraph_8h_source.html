265 template <typename Vertex, typename Edge>
266 typename CompositeGraph<Vertex, Edge>::ED
267 CompositeGraph<Vertex, Edge>::
268 AddEdge(const VID _source, const VID _target, const Edge& _lp) noexcept {
269  if(_source == _target)
270  throw RunTimeException(WHERE) << "Tried to add edge between same VID "
271  << _source << ".";
272 
273  if(_lp.GetWeight() == 0 or _lp.GetWeight() == 0)
274  throw RunTimeException(WHERE) << "Tried to add zero weight edge!";
275 
276  // We need to adjust _lp, but we still want to override the base class
277  // function, so make a local copy of the edge.
278  Edge edge = _lp;
279 
280  // Vector of edge descriptors, which are edges already in individual graphs
281  std::vector<ED>& edgeDescriptors = edge.GetEdgeDescriptors();
282 
283  const Vertex& sourceCfg = this->GetVertex(_source),
284  & targetCfg = this->GetVertex(_target);
285 
286  size_t numInactiveRobots = 0;
287 
288  // First, make sure all the local edges are in the individual graphs.
289  for(size_t i = 0; i < edgeDescriptors.size(); ++i) {
290  auto graph = m_graphs[i];
291 
292  const VID individualSourceVID = sourceCfg.GetVID(i),
293  individualTargetVID = targetCfg.GetVID(i);
294 
295  // If they are the same, it means this is an inactive robot. Record the
296  // number of these that occur so that we can ensure SOME robot(s) moved.
297  if(individualSourceVID == individualTargetVID) {
298  ++numInactiveRobots;
299  // continue;
300  }
301 
302  // Assert that the individual vertices exist.
303  const bool verticesExist =
304  graph->find_vertex(individualSourceVID) != graph->end() and
305  graph->find_vertex(individualTargetVID) != graph->end();
306  if(!verticesExist)
307  throw RunTimeException(WHERE, "Cannot add edge for non-existent vertices.");
308 
309  // If we received a valid edge descriptor, it must agree with the individual
310  // VIDs, and the individual edge must exist.
311  const bool edgeExists = graph->IsEdge(individualSourceVID,
312  individualTargetVID);
313  if(edgeDescriptors[i] != INVALID_ED) {
314  if(!edgeExists)
315  throw RunTimeException(WHERE) << "Expected edge (" << individualSourceVID
316  << ", " << individualTargetVID << ") does "
317  << "not exist for robot "
318  << graph->GetRobot()->GetLabel() << ".";
319 
320  const bool consistent = edgeDescriptors[i].source() == individualSourceVID
321  and edgeDescriptors[i].target() == individualTargetVID;
322  if(!consistent)
323  throw RunTimeException(WHERE) << "Edge descriptors are not consistent. "
324  << "Fetched from group cfg: ("
325  << individualSourceVID << ", "
326  << individualTargetVID << "), fetched from "
327  << "group edge: ("
328  << edgeDescriptors[i].source() << ", "
329  << edgeDescriptors[i].target() << ").";
330  }
331  // If not, assert that the edge to be added does not already exist and then
332  // add it.
333  else
334  throw RunTimeException(WHERE) << "Expected edge (" << individualSourceVID
335  << ", " << individualTargetVID << ") does "
336  << "not exist for robot "
337  << graph->GetRobot()->GetLabel() << ".";
338  }
339 
340  if(numInactiveRobots >= GetNumRobots())
341  throw RunTimeException(WHERE) << "No robots were moved in this edge!";
342 
343  const auto edgeDescriptor = this->add_edge(_source, _target, edge);
344  const bool notNew = edgeDescriptor.id() == INVALID_EID;
345 
346  if(notNew) {
347  std::cerr << "\nCompositeGraph::AddEdge: edge (" << _source << ", "
348  << _target << ") already exists, not adding."
349  << std::endl;
350  }
351  else {
352  // Find the edge iterator.
353  VI vi;
354  EI ei;
355  this->find_edge(edgeDescriptor, vi, ei);
356 
357  // Execute post-add hooks.
358  this->ExecuteAddEdgeHooks(ei);
359 
360  this->m_predecessors[_target].insert(_source);
361  ++m_timestamp;
362  }
363 
364  return edgeDescriptor;
365 }

368 template <typename Vertex, typename Edge>
369 typename CompositeGraph<Vertex, Edge>::VID
370 CompositeGraph<Vertex, Edge>::
371 AddVertex(const Vertex& _v) noexcept {
372  Vertex cfg; // Will be a copy of the const Vertex
373  // Check that the group map is correct.
374  if((CompositeGraphType*)_v.GetGroupGraph() != this) {
375  throw RunTimeException(WHERE) << "Composite Graph of vertex does not "
376  << "match this composite graph.";
377  }
378  else { // Graphs match, no change to attempt.
379  cfg = _v;
380  }
381 
382  // Find the vertex and ensure it does not already exist.
383  CVI vi;
384  if(this->IsVertex(cfg, vi)) {
385  std::cerr << "\nCompositeGraph::AddVertex: vertex " << vi->descriptor()
386  << " already in graph"
387  << std::endl;
388  return vi->descriptor();
389  }
390 
391  // Add each vid to individual graphs if not already present.
392  for(size_t i = 0; i < m_group->Size(); ++i) {
393  auto graph = GetIndividualGraph(i);
394  auto robot = graph->GetRobot();
395  const VID individualVID = cfg.GetVID(i);
396 
397  // If the vid is invalid, we must add the local cfg.
398  if(individualVID == INVALID_VID)
399  cfg.SetRobotCfg(robot, graph->AddVertex(cfg.GetRobotCfg(i)));
400  // If the vid was valid, make sure it exists.
401  else if(graph->find_vertex(individualVID) == graph->end())
402  throw RunTimeException(WHERE) << "Individual vertex " << individualVID
403  << " does not exist!";
404  }
405 
406  // The vertex does not exist. Add it now.
407  const VID vid = this->add_vertex(cfg);
408  this->m_predecessors[vid];
409  this->m_allVIDs.insert(vid);
410  ++m_timestamp;
411 
412  // Execute post-add hooks.
413  this->ExecuteAddVertexHooks(this->find_vertex(vid));
414 
415  return vid;
416 }

419 template <typename Vertex, typename Edge>
420 void
421 CompositeGraph<Vertex, Edge>::
422 DeleteVertex(const VID _v) noexcept {
423  // Find the vertex and crash if it doesn't exist.
424  VI vi = this->find_vertex(_v);
425  if(vi == this->end())
426  throw RunTimeException(WHERE, "VID " + std::to_string(_v) +
427  " is not in the graph.");
428 
429  // We must manually delete the edges rather than letting STAPL do this in
430  // order to call the DeleteEdge hooks.
431 
432  // Delete the outbound edges.
433  for(auto edge = vi->begin(); edge != vi->end(); edge = vi->begin())
434  DeleteEdge(edge);
435 
436  // Delete the inbound edges.
442  for(VI vertex = this->begin(); vertex != this->end(); ++vertex) {
443  while(true) {
444  // Jump to the next edge that targets this vertex.
445  EI edge = stapl::graph_find(vertex->begin(), vertex->end(),
446  stapl::eq_target<VID>(_v));
447  if(edge == vertex->end())
448  break;
449 
450  // If we found one, delete it.
451  DeleteEdge(edge);
452  }
453  }
454 
455  // Execute pre-delete hooks and update vizmo debug.
456  this->ExecuteDeleteVertexHooks(vi);
457 
458  // Delete the group vertex.
459  this->delete_vertex(vi->descriptor());
460  this->m_allVIDs.erase(_v);
461  ++m_timestamp;
462 }

484 DeleteEdge(EI _iterator) noexcept {

486  this->ExecuteDeleteEdgeHooks(_iterator);
487

493

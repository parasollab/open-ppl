593 template <typename MPTraits>
594 double
595 SIPPMethod<MPTraits>::
596 PathWeight(typename SIPPGraph::adj_edge_iterator& _ei,
597  const double _sourceDistance, const double _targetDistance) {
598  //const double timeRes = this->GetEnvironment()->GetTimeRes();
599  const size_t buffer = 2;
600  const size_t zero = 0;
601 
602  size_t sourceDistance = size_t(_sourceDistance);
603 
604  auto source = m_sippGraph->GetVertex(_ei->source());
605  auto target = m_sippGraph->GetVertex(_ei->target());
606 
607  std::vector<Range<size_t>> intervals;
608  if(!m_edgeIntervals.empty()) {
609  intervals = m_edgeIntervals[std::make_pair(source.vid,target.vid)];
610  }
611  else {
612  intervals.push_back(Range<size_t>(zero,SIZE_MAX));
613  }
614 
615  //bool reachedFirstInterval = false;
616  for(auto iter = intervals.begin(); iter != intervals.end(); iter++) {
617  //if(!reachedFirstInterval and *iter != edge.interval) {
618  // continue;
619  //}
620  //reachedFirstInterval = true;
621 
622  auto interval = *iter;
623 
624  // Check if transition is valid w.r.t. intervals
625  size_t minWaitTime = std::max(zero,std::min(interval.min - sourceDistance,interval.min));
626  if(minWaitTime > 0)
627  minWaitTime += buffer;
628  const size_t minStartTime = sourceDistance + minWaitTime;
629 
630  // Check source interval for min start time
631  if(!source.interval.Contains(std::max(std::min(minStartTime-buffer,minStartTime),m_startTime))
632  or !source.interval.Contains(std::max(minStartTime,minStartTime+buffer))) {
633  continue;
634  }
635 
636  // Check edge interval for min start time
637  if(!interval.Contains(std::max(std::min(minStartTime-buffer,minStartTime),m_startTime))
638  or !interval.Contains(std::max(minStartTime,minStartTime+buffer))) {
639  continue;
640  }
641 
642  // Check target interval for overlap
643  const size_t duration = this->GetGroupTask()
644  ? this->GetGroupRoadmap()->GetEdge(source.vid,target.vid).GetTimeSteps()
645  : this->GetRoadmap()->GetEdge(source.vid,target.vid).GetTimeSteps();
646  const size_t minEndTime = minStartTime + duration;
647 
648  size_t waitTime = minWaitTime;
649  if(!target.interval.Contains(std::max(std::min(minEndTime-buffer,minEndTime),m_startTime))
650  or !target.interval.Contains(std::max(minEndTime+buffer,minEndTime))) {
651  // Check that min end time is not greater than the interval range
652  if(target.interval.max < std::max(minEndTime+buffer,minEndTime)) {
653  continue;
654  }
655 
656  waitTime = target.interval.min - minEndTime;
657  if(waitTime > 0)
658  waitTime += buffer;
659 
660  const size_t startTime = minStartTime + waitTime;
661 
662  // Check source interval for start time
663  if(!source.interval.Contains(std::max(std::min(startTime-buffer,startTime),m_startTime))
664  or !source.interval.Contains(std::max(startTime+buffer,startTime))) {
665  continue;
666  }
667 
668  // Check edge interval for start time
669  if(!interval.Contains(std::max(std::min(startTime-buffer,startTime),m_startTime))
670  or !interval.Contains(std::max(startTime+buffer,startTime))) {
671  continue;
672  }
673 
674  if(!target.interval.Contains(std::max(std::min(startTime+duration-buffer,startTime+duration),m_startTime))
675  or !target.interval.Contains(std::max(startTime+duration+buffer,startTime+duration)))
676  continue;
677 
678  waitTime = startTime - sourceDistance;
679  }
680  m_waitTimesteps[_ei->source()][_ei->target()] = waitTime; //size_t(std::ceil(waitTime/timeRes));
681 
682  double edgeCost;
683 
684  if(m_minTime) {
685  edgeCost = double(waitTime + duration);
686  }
687  else {
688  throw RunTimeException(WHERE) << "Arbitrary distance metric based eval not supported.";
689  auto source = m_sippGraph->GetVertex(_ei->source()).vid;
690  auto target = m_sippGraph->GetVertex(_ei->target()).vid;
691  edgeCost = this->GetGroupTask() ? this->GetGroupRoadmap()->GetEdge(source,target).GetWeight()
692  : this->GetRoadmap()->GetEdge(source,target).GetWeight();
693  }
694 
695  return sourceDistance + edgeCost;
696  }
697 
698  return std::numeric_limits<double>::infinity();
699 }
